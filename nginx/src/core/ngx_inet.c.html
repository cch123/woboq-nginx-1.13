17in_addr_t
18ngx_inet_addr(u_char *text, size_t len)
19{
20 u_char *p, c;
21 in_addr_t addr;
22 ngx_uint_t octet, n;
23
24 addr = 0;
25 octet = 0;
26 n = 0;
27
28 for (p = text; p < text + len; p++) {
29 c = *p;
30
31 if (c >= '0' && c <= '9') {
32 octet = octet * 10 + (c - '0');
33
34 if (octet > 255) {
35 return INADDR_NONE;
36 }
37
38 continue;
39 }
40
41 if (c == '.') {
42 addr = (addr << 8) + octet;
43 octet = 0;
44 n++;
45 continue;
46 }
47
48 return INADDR_NONE;
49 }
50
51 if (n == 3) {
52 addr = (addr << 8) + octet;
53 return htonl(addr);
54 }
55
56 return INADDR_NONE;
57}

60#if (NGX_HAVE_INET6)
61
62ngx_int_t
63ngx_inet6_addr(u_char *p, size_t len, u_char *addr)
64{
65 u_char c, *zero, *digit, *s, *d;
66 size_t len4;
67 ngx_uint_t n, nibbles, word;
68
69 if (len == 0) {
70 return NGX_ERROR;
71 }
72
73 zero = NULL;
74 digit = NULL;
75 len4 = 0;
76 nibbles = 0;
77 word = 0;
78 n = 8;
79
80 if (p[0] == ':') {
81 p++;
82 len--;
83 }
84
85 for (/* void */; len; len--) {
86 c = *p++;
87
88 if (c == ':') {
89 if (nibbles) {
90 digit = p;
91 len4 = len;
92 *addr++ = (u_char) (word >> 8);
93 *addr++ = (u_char) (word & 0xff);
94
95 if (--n) {
96 nibbles = 0;
97 word = 0;
98 continue;
99 }
100
101 } else {
102 if (zero == NULL) {
103 digit = p;
104 len4 = len;
105 zero = addr;
106 continue;
107 }
108 }
109
110 return NGX_ERROR;
111 }
112
113 if (c == '.' && nibbles) {
114 if (n < 2 || digit == NULL) {
115 return NGX_ERROR;
116 }
117
118 word = ngx_inet_addr(digit, len4 - 1);
119 if (word == INADDR_NONE) {
120 return NGX_ERROR;
121 }
122
123 word = ntohl(word);
124 *addr++ = (u_char) ((word >> 24) & 0xff);
125 *addr++ = (u_char) ((word >> 16) & 0xff);
126 n--;
127 break;
128 }
129
130 if (++nibbles > 4) {
131 return NGX_ERROR;
132 }
133
134 if (c >= '0' && c <= '9') {
135 word = word * 16 + (c - '0');
136 continue;
137 }
138
139 c |= 0x20;
140
141 if (c >= 'a' && c <= 'f') {
142 word = word * 16 + (c - 'a') + 10;
143 continue;
144 }
145
146 return NGX_ERROR;
147 }
148
149 if (nibbles == 0 && zero == NULL) {
150 return NGX_ERROR;
151 }
152
153 *addr++ = (u_char) (word >> 8);
154 *addr++ = (u_char) (word & 0xff);
155
156 if (--n) {
157 if (zero) {
158 n *= 2;
159 s = addr - 1;
160 d = s + n;
161 while (s >= zero) {
162 *d-- = *s--;
163 }
164 ngx_memzero(zero, n);
165 return NGX_OK;
166 }
167
168 } else {
169 if (zero == NULL) {
170 return NGX_OK;
171 }
172 }
173
174 return NGX_ERROR;
175}

180size_t
181ngx_sock_ntop(struct sockaddr *sa, socklen_t socklen, u_char *text, size_t len,
182 ngx_uint_t port)
183{
184 u_char *p;
185#if (NGX_HAVE_INET6 || NGX_HAVE_UNIX_DOMAIN)
186 size_t n;
187#endif
188 struct sockaddr_in *sin;
189#if (NGX_HAVE_INET6)
190 struct sockaddr_in6 *sin6;
191#endif
192#if (NGX_HAVE_UNIX_DOMAIN)
193 struct sockaddr_un *saun;
194#endif
195
196 switch (sa->sa_family) {
197
198 case AF_INET:
199
200 sin = (struct sockaddr_in *) sa;
201 p = (u_char *) &sin->sin_addr;
202
203 if (port) {
204 p = ngx_snprintf(text, len, "%ud.%ud.%ud.%ud:%d",
205 p[0], p[1], p[2], p[3], ntohs(sin->sin_port));
206 } else {
207 p = ngx_snprintf(text, len, "%ud.%ud.%ud.%ud",
208 p[0], p[1], p[2], p[3]);
209 }
210
211 return (p - text);
212
213#if (NGX_HAVE_INET6)
214
215 case AF_INET6:
216
217 sin6 = (struct sockaddr_in6 *) sa;
218
219 n = 0;
220
221 if (port) {
222 text[n++] = '[';
223 }
224
225 n = ngx_inet6_ntop(sin6->sin6_addr.s6_addr, &text[n], len);
226
227 if (port) {
228 n = ngx_sprintf(&text[1 + n], "]:%d",
229 ntohs(sin6->sin6_port)) - text;
230 }
231
232 return n;
233#endif
234
235#if (NGX_HAVE_UNIX_DOMAIN)
236
237 case AF_UNIX:
238 saun = (struct sockaddr_un *) sa;
239
240 /* on Linux sockaddr might not include sun_path at all */
241
242 if (socklen <= (socklen_t) offsetof(struct sockaddr_un, sun_path)) {
243 p = ngx_snprintf(text, len, "unix:%Z");
244
245 } else {
246 n = ngx_strnlen((u_char *) saun->sun_path,
247 socklen - offsetof(struct sockaddr_un, sun_path));
248 p = ngx_snprintf(text, len, "unix:%*s%Z", n, saun->sun_path);
249 }
250
251 /* we do not include trailing zero in address length */
252
253 return (p - text - 1);
254
255#endif
256
257 default:
258 return 0;
259 }
260}

263size_t
264ngx_inet_ntop(int family, void *addr, u_char *text, size_t len)
265{
266 u_char *p;
267
268 switch (family) {
269
270 case AF_INET:
271
272 p = addr;
273
274 return ngx_snprintf(text, len, "%ud.%ud.%ud.%ud",
275 p[0], p[1], p[2], p[3])
276 - text;
277
278#if (NGX_HAVE_INET6)
279
280 case AF_INET6:
281 return ngx_inet6_ntop(addr, text, len);
282
283#endif
284
285 default:
286 return 0;
287 }
288}

291#if (NGX_HAVE_INET6)
292
293size_t
294ngx_inet6_ntop(u_char *p, u_char *text, size_t len)
295{
296 u_char *dst;
297 size_t max, n;
298 ngx_uint_t i, zero, last;
299
300 if (len < NGX_INET6_ADDRSTRLEN) {
301 return 0;
302 }
303
304 zero = (ngx_uint_t) -1;
305 last = (ngx_uint_t) -1;
306 max = 1;
307 n = 0;
308
309 for (i = 0; i < 16; i += 2) {
310
311 if (p[i] || p[i + 1]) {
312
313 if (max < n) {
314 zero = last;
315 max = n;
316 }
317
318 n = 0;
319 continue;
320 }
321
322 if (n++ == 0) {
323 last = i;
324 }
325 }
326
327 if (max < n) {
328 zero = last;
329 max = n;
330 }
331
332 dst = text;
333 n = 16;
334
335 if (zero == 0) {
336
337 if ((max == 5 && p[10] == 0xff && p[11] == 0xff)
338 || (max == 6)
339 || (max == 7 && p[14] != 0 && p[15] != 1))
340 {
341 n = 12;
342 }
343
344 *dst++ = ':';
345 }
346
347 for (i = 0; i < n; i += 2) {
348
349 if (i == zero) {
350 *dst++ = ':';
351 i += (max - 1) * 2;
352 continue;
353 }
354
355 dst = ngx_sprintf(dst, "%xd", p[i] * 256 + p[i + 1]);
356
357 if (i < 14) {
358 *dst++ = ':';
359 }
360 }
361
362 if (n == 12) {
363 dst = ngx_sprintf(dst, "%ud.%ud.%ud.%ud", p[12], p[13], p[14], p[15]);
364 }
365
366 return dst - text;
367}

372ngx_int_t
373ngx_ptocidr(ngx_str_t *text, ngx_cidr_t *cidr)
374{
375 u_char *addr, *mask, *last;
376 size_t len;
377 ngx_int_t shift;
378#if (NGX_HAVE_INET6)
379 ngx_int_t rc;
380 ngx_uint_t s, i;
381#endif
382
383 addr = text->data;
384 last = addr + text->len;
385
386 mask = ngx_strlchr(addr, last, '/');
387 len = (mask ? mask : last) - addr;
388
389 cidr->u.in.addr = ngx_inet_addr(addr, len);
390
391 if (cidr->u.in.addr != INADDR_NONE) {
392 cidr->family = AF_INET;
393
394 if (mask == NULL) {
395 cidr->u.in.mask = 0xffffffff;
396 return NGX_OK;
397 }
398
399#if (NGX_HAVE_INET6)
400 } else if (ngx_inet6_addr(addr, len, cidr->u.in6.addr.s6_addr) == NGX_OK) {
401 cidr->family = AF_INET6;
402
403 if (mask == NULL) {
404 ngx_memset(cidr->u.in6.mask.s6_addr, 0xff, 16);
405 return NGX_OK;
406 }
407
408#endif
409 } else {
410 return NGX_ERROR;
411 }
412
413 mask++;
414
415 shift = ngx_atoi(mask, last - mask);
416 if (shift == NGX_ERROR) {
417 return NGX_ERROR;
418 }
419
420 switch (cidr->family) {
421
422#if (NGX_HAVE_INET6)
423 case AF_INET6:
424 if (shift > 128) {
425 return NGX_ERROR;
426 }
427
428 addr = cidr->u.in6.addr.s6_addr;
429 mask = cidr->u.in6.mask.s6_addr;
430 rc = NGX_OK;
431
432 for (i = 0; i < 16; i++) {
433
434 s = (shift > 8) ? 8 : shift;
435 shift -= s;
436
437 mask[i] = (u_char) (0xffu << (8 - s));
438
439 if (addr[i] != (addr[i] & mask[i])) {
440 rc = NGX_DONE;
441 addr[i] &= mask[i];
442 }
443 }
444
445 return rc;
446#endif
447
448 default: /* AF_INET */
449 if (shift > 32) {
450 return NGX_ERROR;
451 }
452
453 if (shift) {
454 cidr->u.in.mask = htonl((uint32_t) (0xffffffffu << (32 - shift)));
455
456 } else {
457 /* x86 compilers use a shl instruction that shifts by modulo 32 */
458 cidr->u.in.mask = 0;
459 }
460
461 if (cidr->u.in.addr == (cidr->u.in.addr & cidr->u.in.mask)) {
462 return NGX_OK;
463 }
464
465 cidr->u.in.addr &= cidr->u.in.mask;
466
467 return NGX_DONE;
468 }
469}

472ngx_int_t
473ngx_cidr_match(struct sockaddr *sa, ngx_array_t *cidrs)
474{
475#if (NGX_HAVE_INET6)
476 u_char *p;
477#endif
478 in_addr_t inaddr;
479 ngx_cidr_t *cidr;
480 ngx_uint_t family, i;
481#if (NGX_HAVE_INET6)
482 ngx_uint_t n;
483 struct in6_addr *inaddr6;
484#endif
485
486#if (NGX_SUPPRESS_WARN)
487 inaddr = 0;
488#if (NGX_HAVE_INET6)
489 inaddr6 = NULL;
490#endif
491#endif
492
493 family = sa->sa_family;
494
495 if (family == AF_INET) {
496 inaddr = ((struct sockaddr_in *) sa)->sin_addr.s_addr;
497 }
498
499#if (NGX_HAVE_INET6)
500 else if (family == AF_INET6) {
501 inaddr6 = &((struct sockaddr_in6 *) sa)->sin6_addr;
502
503 if (IN6_IS_ADDR_V4MAPPED(inaddr6)) {
504 family = AF_INET;
505
506 p = inaddr6->s6_addr;
507
508 inaddr = p[12] << 24;
509 inaddr += p[13] << 16;
510 inaddr += p[14] << 8;
511 inaddr += p[15];
512
513 inaddr = htonl(inaddr);
514 }
515 }
516#endif
517
518 for (cidr = cidrs->elts, i = 0; i < cidrs->nelts; i++) {
519 if (cidr[i].family != family) {
520 goto next;
521 }
522
523 switch (family) {
524
525#if (NGX_HAVE_INET6)
526 case AF_INET6:
527 for (n = 0; n < 16; n++) {
528 if ((inaddr6->s6_addr[n] & cidr[i].u.in6.mask.s6_addr[n])
529 != cidr[i].u.in6.addr.s6_addr[n])
530 {
531 goto next;
532 }
533 }
534 break;
535#endif
536
537#if (NGX_HAVE_UNIX_DOMAIN)
538 case AF_UNIX:
539 break;
540#endif
541
542 default: /* AF_INET */
543 if ((inaddr & cidr[i].u.in.mask) != cidr[i].u.in.addr) {
544 goto next;
545 }
546 break;
547 }
548
549 return NGX_OK;
550
551 next:
552 continue;
553 }
554
555 return NGX_DECLINED;
556}

559ngx_int_t
560ngx_parse_addr(ngx_pool_t *pool, ngx_addr_t *addr, u_char *text, size_t len)
561{
562 in_addr_t inaddr;
563 ngx_uint_t family;
564 struct sockaddr_in *sin;
565#if (NGX_HAVE_INET6)
566 struct in6_addr inaddr6;
567 struct sockaddr_in6 *sin6;
568
569 /*
570 * prevent MSVC8 warning:
571 * potentially uninitialized local variable 'inaddr6' used
572 */
573 ngx_memzero(&inaddr6, sizeof(struct in6_addr));
574#endif
575
576 inaddr = ngx_inet_addr(text, len);
577
578 if (inaddr != INADDR_NONE) {
579 family = AF_INET;
580 len = sizeof(struct sockaddr_in);
581
582#if (NGX_HAVE_INET6)
583 } else if (ngx_inet6_addr(text, len, inaddr6.s6_addr) == NGX_OK) {
584 family = AF_INET6;
585 len = sizeof(struct sockaddr_in6);
586
587#endif
588 } else {
589 return NGX_DECLINED;
590 }
591
592 addr->sockaddr = ngx_pcalloc(pool, len);
593 if (addr->sockaddr == NULL) {
594 return NGX_ERROR;
595 }
596
597 addr->sockaddr->sa_family = (u_char) family;
598 addr->socklen = len;
599
600 switch (family) {
601
602#if (NGX_HAVE_INET6)
603 case AF_INET6:
604 sin6 = (struct sockaddr_in6 *) addr->sockaddr;
605 ngx_memcpy(sin6->sin6_addr.s6_addr, inaddr6.s6_addr, 16);
606 break;
607#endif
608
609 default: /* AF_INET */
610 sin = (struct sockaddr_in *) addr->sockaddr;
611 sin->sin_addr.s_addr = inaddr;
612 break;
613 }
614
615 return NGX_OK;
616}

619ngx_int_t
620ngx_parse_addr_port(ngx_pool_t *pool, ngx_addr_t *addr, u_char *text,
621 size_t len)
622{
623 u_char *p, *last;
624 size_t plen;
625 ngx_int_t rc, port;
626
627 rc = ngx_parse_addr(pool, addr, text, len);
628
629 if (rc != NGX_DECLINED) {
630 return rc;
631 }
632
633 last = text + len;
634
635#if (NGX_HAVE_INET6)
636 if (len && text[0] == '[') {
637
638 p = ngx_strlchr(text, last, ']');
639
640 if (p == NULL || p == last - 1 || *++p != ':') {
641 return NGX_DECLINED;
642 }
643
644 text++;
645 len -= 2;
646
647 } else
648#endif
649
650 {
651 p = ngx_strlchr(text, last, ':');
652
653 if (p == NULL) {
654 return NGX_DECLINED;
655 }
656 }
657
658 p++;
659 plen = last - p;
660
661 port = ngx_atoi(p, plen);
662
663 if (port < 1 || port > 65535) {
664 return NGX_DECLINED;
665 }
666
667 len -= plen + 1;
668
669 rc = ngx_parse_addr(pool, addr, text, len);
670
671 if (rc != NGX_OK) {
672 return rc;
673 }
674
675 ngx_inet_set_port(addr->sockaddr, (in_port_t) port);
676
677 return NGX_OK;
678}

681ngx_int_t
682ngx_parse_url(ngx_pool_t *pool, ngx_url_t *u)
683{
684 u_char *p;
685 size_t len;
686
687 p = u->url.data;
688 len = u->url.len;
689
690 if (len >= 5 && ngx_strncasecmp(p, (u_char *) "unix:", 5) == 0) {
691 return ngx_parse_unix_domain_url(pool, u);
692 }
693
694 if (len && p[0] == '[') {
695 return ngx_parse_inet6_url(pool, u);
696 }
697
698 return ngx_parse_inet_url(pool, u);
699}

702static ngx_int_t
703ngx_parse_unix_domain_url(ngx_pool_t *pool, ngx_url_t *u)
704{
705#if (NGX_HAVE_UNIX_DOMAIN)
706 u_char *path, *uri, *last;
707 size_t len;
708 struct sockaddr_un *saun;
709
710 len = u->url.len;
711 path = u->url.data;
712
713 path += 5;
714 len -= 5;
715
716 if (u->uri_part) {
717
718 last = path + len;
719 uri = ngx_strlchr(path, last, ':');
720
721 if (uri) {
722 len = uri - path;
723 uri++;
724 u->uri.len = last - uri;
725 u->uri.data = uri;
726 }
727 }
728
729 if (len == 0) {
730 u->err = "no path in the unix domain socket";
731 return NGX_ERROR;
732 }
733
734 u->host.len = len++;
735 u->host.data = path;
736
737 if (len > sizeof(saun->sun_path)) {
738 u->err = "too long path in the unix domain socket";
739 return NGX_ERROR;
740 }
741
742 u->socklen = sizeof(struct sockaddr_un);
743 saun = (struct sockaddr_un *) &u->sockaddr;
744 saun->sun_family = AF_UNIX;
745 (void) ngx_cpystrn((u_char *) saun->sun_path, path, len);
746
747 u->addrs = ngx_pcalloc(pool, sizeof(ngx_addr_t));
748 if (u->addrs == NULL) {
749 return NGX_ERROR;
750 }
751
752 saun = ngx_pcalloc(pool, sizeof(struct sockaddr_un));
753 if (saun == NULL) {
754 return NGX_ERROR;
755 }
756
757 u->family = AF_UNIX;
758 u->naddrs = 1;
759
760 saun->sun_family = AF_UNIX;
761 (void) ngx_cpystrn((u_char *) saun->sun_path, path, len);
762
763 u->addrs[0].sockaddr = (struct sockaddr *) saun;
764 u->addrs[0].socklen = sizeof(struct sockaddr_un);
765 u->addrs[0].name.len = len + 4;
766 u->addrs[0].name.data = u->url.data;
767
768 return NGX_OK;
769
770#else
771
772 u->err = "the unix domain sockets are not supported on this platform";
773
774 return NGX_ERROR;
775
776#endif
777}

780static ngx_int_t
781ngx_parse_inet_url(ngx_pool_t *pool, ngx_url_t *u)
782{
783 u_char *p, *host, *port, *last, *uri, *args;
784 size_t len;
785 ngx_int_t n;
786 struct sockaddr_in *sin;
787#if (NGX_HAVE_INET6)
788 struct sockaddr_in6 *sin6;
789#endif
790
791 u->socklen = sizeof(struct sockaddr_in);
792 sin = (struct sockaddr_in *) &u->sockaddr;
793 sin->sin_family = AF_INET;
794
795 u->family = AF_INET;
796
797 host = u->url.data;
798
799 last = host + u->url.len;
800
801 port = ngx_strlchr(host, last, ':');
802
803 uri = ngx_strlchr(host, last, '/');
804
805 args = ngx_strlchr(host, last, '?');
806
807 if (args) {
808 if (uri == NULL || args < uri) {
809 uri = args;
810 }
811 }
812
813 if (uri) {
814 if (u->listen || !u->uri_part) {
815 u->err = "invalid host";
816 return NGX_ERROR;
817 }
818
819 u->uri.len = last - uri;
820 u->uri.data = uri;
821
822 last = uri;
823
824 if (uri < port) {
825 port = NULL;
826 }
827 }
828
829 if (port) {
830 port++;
831
832 len = last - port;
833
834 n = ngx_atoi(port, len);
835
836 if (n < 1 || n > 65535) {
837 u->err = "invalid port";
838 return NGX_ERROR;
839 }
840
841 u->port = (in_port_t) n;
842 sin->sin_port = htons((in_port_t) n);
843
844 u->port_text.len = len;
845 u->port_text.data = port;
846
847 last = port - 1;
848
849 } else {
850 if (uri == NULL) {
851
852 if (u->listen) {
853
854 /* test value as port only */
855
856 n = ngx_atoi(host, last - host);
857
858 if (n != NGX_ERROR) {
859
860 if (n < 1 || n > 65535) {
861 u->err = "invalid port";
862 return NGX_ERROR;
863 }
864
865 u->port = (in_port_t) n;
866 sin->sin_port = htons((in_port_t) n);
867
868 u->port_text.len = last - host;
869 u->port_text.data = host;
870
871 u->wildcard = 1;
872
873 return NGX_OK;
874 }
875 }
876 }
877
878 u->no_port = 1;
879 u->port = u->default_port;
880 sin->sin_port = htons(u->default_port);
881 }
882
883 len = last - host;
884
885 if (len == 0) {
886 u->err = "no host";
887 return NGX_ERROR;
888 }
889
890 u->host.len = len;
891 u->host.data = host;
892
893 if (u->listen && len == 1 && *host == '*') {
894 sin->sin_addr.s_addr = INADDR_ANY;
895 u->wildcard = 1;
896 return NGX_OK;
897 }
898
899 sin->sin_addr.s_addr = ngx_inet_addr(host, len);
900
901 if (sin->sin_addr.s_addr != INADDR_NONE) {
902
903 if (sin->sin_addr.s_addr == INADDR_ANY) {
904 u->wildcard = 1;
905 }
906
907 u->naddrs = 1;
908
909 u->addrs = ngx_pcalloc(pool, sizeof(ngx_addr_t));
910 if (u->addrs == NULL) {
911 return NGX_ERROR;
912 }
913
914 sin = ngx_pcalloc(pool, sizeof(struct sockaddr_in));
915 if (sin == NULL) {
916 return NGX_ERROR;
917 }
918
919 ngx_memcpy(sin, &u->sockaddr, sizeof(struct sockaddr_in));
920
921 u->addrs[0].sockaddr = (struct sockaddr *) sin;
922 u->addrs[0].socklen = sizeof(struct sockaddr_in);
923
924 p = ngx_pnalloc(pool, u->host.len + sizeof(":65535") - 1);
925 if (p == NULL) {
926 return NGX_ERROR;
927 }
928
929 u->addrs[0].name.len = ngx_sprintf(p, "%V:%d",
930 &u->host, u->port) - p;
931 u->addrs[0].name.data = p;
932
933 return NGX_OK;
934 }
935
936 if (u->no_resolve) {
937 return NGX_OK;
938 }
939
940 if (ngx_inet_resolve_host(pool, u) != NGX_OK) {
941 return NGX_ERROR;
942 }
943
944 u->family = u->addrs[0].sockaddr->sa_family;
945 u->socklen = u->addrs[0].socklen;
946 ngx_memcpy(&u->sockaddr, u->addrs[0].sockaddr, u->addrs[0].socklen);
947
948 switch (u->family) {
949
950#if (NGX_HAVE_INET6)
951 case AF_INET6:
952 sin6 = (struct sockaddr_in6 *) &u->sockaddr;
953
954 if (IN6_IS_ADDR_UNSPECIFIED(&sin6->sin6_addr)) {
955 u->wildcard = 1;
956 }
957
958 break;
959#endif
960
961 default: /* AF_INET */
962 sin = (struct sockaddr_in *) &u->sockaddr;
963
964 if (sin->sin_addr.s_addr == INADDR_ANY) {
965 u->wildcard = 1;
966 }
967
968 break;
969 }
970
971 return NGX_OK;
972}

976ngx_parse_inet6_url(ngx_pool_t *pool, ngx_url_t *u)
977{
978#if (NGX_HAVE_INET6)
979 u_char *p, *host, *port, *last, *uri;
980 size_t len;
981 ngx_int_t n;
982 struct sockaddr_in6 *sin6;
983
984 u->socklen = sizeof(struct sockaddr_in6);
985 sin6 = (struct sockaddr_in6 *) &u->sockaddr;
986 sin6->sin6_family = AF_INET6;
987
988 host = u->url.data + 1;
989
990 last = u->url.data + u->url.len;
991
992 p = ngx_strlchr(host, last, ']');
993
994 if (p == NULL) {
995 u->err = "invalid host";
996 return NGX_ERROR;
997 }
998
999 port = p + 1;
1000
1001 uri = ngx_strlchr(port, last, '/');
1002
1003 if (uri) {
1004 if (u->listen || !u->uri_part) {
1005 u->err = "invalid host";
1006 return NGX_ERROR;
1007 }
1008
1009 u->uri.len = last - uri;
1010 u->uri.data = uri;
1011
1012 last = uri;
1013 }
1014
1015 if (port < last) {
1016 if (*port != ':') {
1017 u->err = "invalid host";
1018 return NGX_ERROR;
1019 }
1020
1021 port++;
1022
1023 len = last - port;
1024
1025 n = ngx_atoi(port, len);
1026
1027 if (n < 1 || n > 65535) {
1028 u->err = "invalid port";
1029 return NGX_ERROR;
1030 }
1031
1032 u->port = (in_port_t) n;
1033 sin6->sin6_port = htons((in_port_t) n);
1034
1035 u->port_text.len = len;
1036 u->port_text.data = port;
1037
1038 } else {
1039 u->no_port = 1;
1040 u->port = u->default_port;
1041 sin6->sin6_port = htons(u->default_port);
1042 }
1043
1044 len = p - host;
1045
1046 if (len == 0) {
1047 u->err = "no host";
1048 return NGX_ERROR;
1049 }
1050
1051 u->host.len = len + 2;
1052 u->host.data = host - 1;
1053
1054 if (ngx_inet6_addr(host, len, sin6->sin6_addr.s6_addr) != NGX_OK) {
1055 u->err = "invalid IPv6 address";
1056 return NGX_ERROR;
1057 }
1058
1059 if (IN6_IS_ADDR_UNSPECIFIED(&sin6->sin6_addr)) {
1060 u->wildcard = 1;
1061 }
1062
1063 u->family = AF_INET6;
1064 u->naddrs = 1;
1065
1066 u->addrs = ngx_pcalloc(pool, sizeof(ngx_addr_t));
1067 if (u->addrs == NULL) {
1068 return NGX_ERROR;
1069 }
1070
1071 sin6 = ngx_pcalloc(pool, sizeof(struct sockaddr_in6));
1072 if (sin6 == NULL) {
1073 return NGX_ERROR;
1074 }
1075
1076 ngx_memcpy(sin6, &u->sockaddr, sizeof(struct sockaddr_in6));
1077
1078 u->addrs[0].sockaddr = (struct sockaddr *) sin6;
1079 u->addrs[0].socklen = sizeof(struct sockaddr_in6);
1080
1081 p = ngx_pnalloc(pool, u->host.len + sizeof(":65535") - 1);
1082 if (p == NULL) {
1083 return NGX_ERROR;
1084 }
1085
1086 u->addrs[0].name.len = ngx_sprintf(p, "%V:%d",
1087 &u->host, u->port) - p;
1088 u->addrs[0].name.data = p;
1089
1090 return NGX_OK;
1091
1092#else
1093
1094 u->err = "the INET6 sockets are not supported on this platform";
1095
1096 return NGX_ERROR;
1097
1098#endif
1099}

1104ngx_int_t
1105ngx_inet_resolve_host(ngx_pool_t *pool, ngx_url_t *u)
1106{
1107 u_char *p, *host;
1108 size_t len;
1109 in_port_t port;
1110 ngx_uint_t i;
1111 struct addrinfo hints, *res, *rp;
1112 struct sockaddr_in *sin;
1113 struct sockaddr_in6 *sin6;
1114
1115 port = htons(u->port);
1116
1117 host = ngx_alloc(u->host.len + 1, pool->log);
1118 if (host == NULL) {
1119 return NGX_ERROR;
1120 }
1121
1122 (void) ngx_cpystrn(host, u->host.data, u->host.len + 1);
1123
1124 ngx_memzero(&hints, sizeof(struct addrinfo));
1125 hints.ai_family = AF_UNSPEC;
1126 hints.ai_socktype = SOCK_STREAM;
1127#ifdef AI_ADDRCONFIG
1128 hints.ai_flags = AI_ADDRCONFIG;
1129#endif
1130
1131 if (getaddrinfo((char *) host, NULL, &hints, &res) != 0) {
1132 u->err = "host not found";
1133 ngx_free(host);
1134 return NGX_ERROR;
1135 }
1136
1137 ngx_free(host);
1138
1139 for (i = 0, rp = res; rp != NULL; rp = rp->ai_next) {
1140
1141 switch (rp->ai_family) {
1142
1143 case AF_INET:
1144 case AF_INET6:
1145 break;
1146
1147 default:
1148 continue;
1149 }
1150
1151 i++;
1152 }
1153
1154 if (i == 0) {
1155 u->err = "host not found";
1156 goto failed;
1157 }
1158
1159 /* MP: ngx_shared_palloc() */
1160
1161 u->addrs = ngx_pcalloc(pool, i * sizeof(ngx_addr_t));
1162 if (u->addrs == NULL) {
1163 goto failed;
1164 }
1165
1166 u->naddrs = i;
1167
1168 i = 0;
1169
1170 /* AF_INET addresses first */
1171
1172 for (rp = res; rp != NULL; rp = rp->ai_next) {
1173
1174 if (rp->ai_family != AF_INET) {
1175 continue;
1176 }
1177
1178 sin = ngx_pcalloc(pool, rp->ai_addrlen);
1179 if (sin == NULL) {
1180 goto failed;
1181 }
1182
1183 ngx_memcpy(sin, rp->ai_addr, rp->ai_addrlen);
1184
1185 sin->sin_port = port;
1186
1187 u->addrs[i].sockaddr = (struct sockaddr *) sin;
1188 u->addrs[i].socklen = rp->ai_addrlen;
1189
1190 len = NGX_INET_ADDRSTRLEN + sizeof(":65535") - 1;
1191
1192 p = ngx_pnalloc(pool, len);
1193 if (p == NULL) {
1194 goto failed;
1195 }
1196
1197 len = ngx_sock_ntop((struct sockaddr *) sin, rp->ai_addrlen, p, len, 1);
1198
1199 u->addrs[i].name.len = len;
1200 u->addrs[i].name.data = p;
1201
1202 i++;
1203 }
1204
1205 for (rp = res; rp != NULL; rp = rp->ai_next) {
1206
1207 if (rp->ai_family != AF_INET6) {
1208 continue;
1209 }
1210
1211 sin6 = ngx_pcalloc(pool, rp->ai_addrlen);
1212 if (sin6 == NULL) {
1213 goto failed;
1214 }
1215
1216 ngx_memcpy(sin6, rp->ai_addr, rp->ai_addrlen);
1217
1218 sin6->sin6_port = port;
1219
1220 u->addrs[i].sockaddr = (struct sockaddr *) sin6;
1221 u->addrs[i].socklen = rp->ai_addrlen;
1222
1223 len = NGX_INET6_ADDRSTRLEN + sizeof("[]:65535") - 1;
1224
1225 p = ngx_pnalloc(pool, len);
1226 if (p == NULL) {
1227 goto failed;
1228 }
1229
1230 len = ngx_sock_ntop((struct sockaddr *) sin6, rp->ai_addrlen, p,
1231 len, 1);
1232
1233 u->addrs[i].name.len = len;
1234 u->addrs[i].name.data = p;
1235
1236 i++;
1237 }
1238
1239 freeaddrinfo(res);
1240 return NGX_OK;
1241
1242failed:
1243
1244 freeaddrinfo(res);
1245 return NGX_ERROR;
1246}

1251ngx_inet_resolve_host(ngx_pool_t *pool, ngx_url_t *u)

1298 if (sin == NULL) {
1299 return NGX_ERROR;
1300 }

1344 u->addrs[0].socklen = sizeof(struct sockaddr_in);
1345

1347 if (p == NULL) {
1348 return NGX_ERROR;

1362ngx_int_t
1363ngx_cmp_sockaddr(struct sockaddr *sa1, socklen_t slen1,
1364 struct sockaddr *sa2, socklen_t slen2, ngx_uint_t cmp_port)
1365{
1366 struct sockaddr_in *sin1, *sin2;
1367#if (NGX_HAVE_INET6)
1368 struct sockaddr_in6 *sin61, *sin62;
1369#endif
1370#if (NGX_HAVE_UNIX_DOMAIN)
1371 size_t len;
1372 struct sockaddr_un *saun1, *saun2;
1373#endif
1374
1375 if (sa1->sa_family != sa2->sa_family) {
1376 return NGX_DECLINED;
1377 }
1378
1379 switch (sa1->sa_family) {
1380
1381#if (NGX_HAVE_INET6)
1382 case AF_INET6:
1383
1384 sin61 = (struct sockaddr_in6 *) sa1;
1385 sin62 = (struct sockaddr_in6 *) sa2;
1386
1387 if (cmp_port && sin61->sin6_port != sin62->sin6_port) {
1388 return NGX_DECLINED;
1389 }
1390
1391 if (ngx_memcmp(&sin61->sin6_addr, &sin62->sin6_addr, 16) != 0) {
1392 return NGX_DECLINED;
1393 }
1394
1395 break;
1396#endif
1397
1398#if (NGX_HAVE_UNIX_DOMAIN)
1399 case AF_UNIX:
1400
1401 saun1 = (struct sockaddr_un *) sa1;
1402 saun2 = (struct sockaddr_un *) sa2;
1403
1404 if (slen1 < slen2) {
1405 len = slen1 - offsetof(struct sockaddr_un, sun_path);
1406
1407 } else {
1408 len = slen2 - offsetof(struct sockaddr_un, sun_path);
1409 }
1410
1411 if (len > sizeof(saun1->sun_path)) {
1412 len = sizeof(saun1->sun_path);
1413 }
1414
1415 if (ngx_memcmp(&saun1->sun_path, &saun2->sun_path, len) != 0) {
1416 return NGX_DECLINED;
1417 }
1418
1419 break;
1420#endif
1421
1422 default: /* AF_INET */
1423
1424 sin1 = (struct sockaddr_in *) sa1;
1425 sin2 = (struct sockaddr_in *) sa2;
1426
1427 if (cmp_port && sin1->sin_port != sin2->sin_port) {
1428 return NGX_DECLINED;
1429 }
1430
1431 if (sin1->sin_addr.s_addr != sin2->sin_addr.s_addr) {
1432 return NGX_DECLINED;
1433 }
1434
1435 break;
1436 }
1437
1438 return NGX_OK;
1439}
