16/* 32-bit crc16 */
17
18static ngx_inline uint32_t
19ngx_crc(u_char *data, size_t len)
20{
21 uint32_t sum;
22
23 for (sum = 0; len; len--) {
24
25 /*
26 * gcc 2.95.2 x86 and icc 7.1.006 compile
27 * that operator into the single "rol" opcode,
28 * msvc 6.0sp2 compiles it into four opcodes.
29 */
30 sum = sum >> 1 | sum << 31;
31
32 sum += *data++;
33 }
34
35 return sum;
36}
