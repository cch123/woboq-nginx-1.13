67static ngx_inline u_char *
68ngx_strlchr(u_char *p, u_char *last, u_char c)
69{
70 while (p < last) {
71
72 if (*p == c) {
73 return p;
74 }
75
76 p++;
77 }
78
79 return NULL;
80}

83/*
84 * msvc and icc7 compile memset() to the inline "rep stos"
85 * while ZeroMemory() and bzero() are the calls.
86 * icc7 may also inline several mov's of a zeroed register for small blocks.
87 */
88#define ngx_memzero(buf, n) (void) memset(buf, 0, n)
89#define ngx_memset(buf, c, n) (void) memset(buf, c, n)

99/*
100 * gcc3, msvc, and icc7 compile memcpy() to the inline "rep movs".
101 * gcc3 compiles memcpy(d, s, 4) to the inline "mov"es.
102 * icc8 compile memcpy(d, s, 4) to the inline "mov"es or XMM moves.
103 */
104#define ngx_memcpy(dst, src, n) (void) memcpy(dst, src, n)
105#define ngx_cpymem(dst, src, n) (((u_char *) memcpy(dst, src, n)) + (n))
106

112/*
113 * the simple inline cycle copies the variable length strings up to 16
114 * bytes faster than icc8 autodetecting _intel_fast_memcpy()
115 */
116
117static ngx_inline u_char *
118ngx_copy(u_char *dst, u_char *src, size_t len)
119{
120 if (len < 17) {
121
122 while (len) {
123 *dst++ = *src++;
124 len--;
125 }
126
127 return dst;
128
129 } else {
130 return ngx_cpymem(dst, src, len);
131 }
132}

135
136#define ngx_copy ngx_cpymem
