35void
36ngx_http_script_flush_complex_value(ngx_http_request_t *r,
37 ngx_http_complex_value_t *val)
38{
39 ngx_uint_t *index;
40
41 index = val->flushes;
42
43 if (index) {
44 while (*index != (ngx_uint_t) -1) {
45
46 if (r->variables[*index].no_cacheable) {
47 r->variables[*index].valid = 0;
48 r->variables[*index].not_found = 0;
49 }
50
51 index++;
52 }
53 }
54}

57ngx_int_t
58ngx_http_complex_value(ngx_http_request_t *r, ngx_http_complex_value_t *val,
59 ngx_str_t *value)
60{
61 size_t len;
62 ngx_http_script_code_pt code;
63 ngx_http_script_len_code_pt lcode;
64 ngx_http_script_engine_t e;
65
66 if (val->lengths == NULL) {
67 *value = val->value;
68 return NGX_OK;
69 }
70
71 ngx_http_script_flush_complex_value(r, val);
72
73 ngx_memzero(&e, sizeof(ngx_http_script_engine_t));
74
75 e.ip = val->lengths;
76 e.request = r;
77 e.flushed = 1;
78
79 len = 0;
80
81 while (*(uintptr_t *) e.ip) {
82 lcode = *(ngx_http_script_len_code_pt *) e.ip;
83 len += lcode(&e);
84 }
85
86 value->len = len;
87 value->data = ngx_pnalloc(r->pool, len);
88 if (value->data == NULL) {
89 return NGX_ERROR;
90 }
91
92 e.ip = val->values;
93 e.pos = value->data;
94 e.buf = *value;
95
96 while (*(uintptr_t *) e.ip) {
97 code = *(ngx_http_script_code_pt *) e.ip;
98 code((ngx_http_script_engine_t *) &e);
99 }
100
101 *value = e.buf;
102
103 return NGX_OK;
104}

107ngx_int_t
108ngx_http_compile_complex_value(ngx_http_compile_complex_value_t *ccv)
109{
110 ngx_str_t *v;
111 ngx_uint_t i, n, nv, nc;
112 ngx_array_t flushes, lengths, values, *pf, *pl, *pv;
113 ngx_http_script_compile_t sc;
114
115 v = ccv->value;
116
117 nv = 0;
118 nc = 0;
119
120 for (i = 0; i < v->len; i++) {
121 if (v->data[i] == '$') {
122 if (v->data[i + 1] >= '1' && v->data[i + 1] <= '9') {
123 nc++;
124
125 } else {
126 nv++;
127 }
128 }
129 }
130
131 if ((v->len == 0 || v->data[0] != '$')
132 && (ccv->conf_prefix || ccv->root_prefix))
133 {
134 if (ngx_conf_full_name(ccv->cf->cycle, v, ccv->conf_prefix) != NGX_OK) {
135 return NGX_ERROR;
136 }
137
138 ccv->conf_prefix = 0;
139 ccv->root_prefix = 0;
140 }
141
142 ccv->complex_value->value = *v;
143 ccv->complex_value->flushes = NULL;
144 ccv->complex_value->lengths = NULL;
145 ccv->complex_value->values = NULL;
146
147 if (nv == 0 && nc == 0) {
148 return NGX_OK;
149 }
150
151 n = nv + 1;
152
153 if (ngx_array_init(&flushes, ccv->cf->pool, n, sizeof(ngx_uint_t))
154 != NGX_OK)
155 {
156 return NGX_ERROR;
157 }
158
159 n = nv * (2 * sizeof(ngx_http_script_copy_code_t)
160 + sizeof(ngx_http_script_var_code_t))
161 + sizeof(uintptr_t);
162
163 if (ngx_array_init(&lengths, ccv->cf->pool, n, 1) != NGX_OK) {
164 return NGX_ERROR;
165 }
166
167 n = (nv * (2 * sizeof(ngx_http_script_copy_code_t)
168 + sizeof(ngx_http_script_var_code_t))
169 + sizeof(uintptr_t)
170 + v->len
171 + sizeof(uintptr_t) - 1)
172 & ~(sizeof(uintptr_t) - 1);
173
174 if (ngx_array_init(&values, ccv->cf->pool, n, 1) != NGX_OK) {
175 return NGX_ERROR;
176 }
177
178 pf = &flushes;
179 pl = &lengths;
180 pv = &values;
181
182 ngx_memzero(&sc, sizeof(ngx_http_script_compile_t));
183
184 sc.cf = ccv->cf;
185 sc.source = v;
186 sc.flushes = &pf;
187 sc.lengths = &pl;
188 sc.values = &pv;
189 sc.complete_lengths = 1;
190 sc.complete_values = 1;
191 sc.zero = ccv->zero;
192 sc.conf_prefix = ccv->conf_prefix;
193 sc.root_prefix = ccv->root_prefix;
194
195 if (ngx_http_script_compile(&sc) != NGX_OK) {
196 return NGX_ERROR;
197 }
198
199 if (flushes.nelts) {
200 ccv->complex_value->flushes = flushes.elts;
201 ccv->complex_value->flushes[flushes.nelts] = (ngx_uint_t) -1;
202 }
203
204 ccv->complex_value->lengths = lengths.elts;
205 ccv->complex_value->values = values.elts;
206
207 return NGX_OK;
208}

211char *
212ngx_http_set_complex_value_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
213{
214 char *p = conf;
215
216 ngx_str_t *value;
217 ngx_http_complex_value_t **cv;
218 ngx_http_compile_complex_value_t ccv;
219
220 cv = (ngx_http_complex_value_t **) (p + cmd->offset);
221
222 if (*cv != NULL) {
223 return "is duplicate";
224 }
225
226 *cv = ngx_palloc(cf->pool, sizeof(ngx_http_complex_value_t));
227 if (*cv == NULL) {
228 return NGX_CONF_ERROR;
229 }
230
231 value = cf->args->elts;
232
233 ngx_memzero(&ccv, sizeof(ngx_http_compile_complex_value_t));
234
235 ccv.cf = cf;
236 ccv.value = &value[1];
237 ccv.complex_value = *cv;
238
239 if (ngx_http_compile_complex_value(&ccv) != NGX_OK) {
240 return NGX_CONF_ERROR;
241 }
242
243 return NGX_CONF_OK;
244}

247ngx_int_t
248ngx_http_test_predicates(ngx_http_request_t *r, ngx_array_t *predicates)
249{
250 ngx_str_t val;
251 ngx_uint_t i;
252 ngx_http_complex_value_t *cv;
253
254 if (predicates == NULL) {
255 return NGX_OK;
256 }
257
258 cv = predicates->elts;
259
260 for (i = 0; i < predicates->nelts; i++) {
261 if (ngx_http_complex_value(r, &cv[i], &val) != NGX_OK) {
262 return NGX_ERROR;
263 }
264
265 if (val.len && (val.len != 1 || val.data[0] != '0')) {
266 return NGX_DECLINED;
267 }
268 }
269
270 return NGX_OK;
271}

274char *
275ngx_http_set_predicate_slot(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
276{
277 char *p = conf;
278
279 ngx_str_t *value;
280 ngx_uint_t i;
281 ngx_array_t **a;
282 ngx_http_complex_value_t *cv;
283 ngx_http_compile_complex_value_t ccv;
284
285 a = (ngx_array_t **) (p + cmd->offset);
286
287 if (*a == NGX_CONF_UNSET_PTR) {
288 *a = ngx_array_create(cf->pool, 1, sizeof(ngx_http_complex_value_t));
289 if (*a == NULL) {
290 return NGX_CONF_ERROR;
291 }
292 }
293
294 value = cf->args->elts;
295
296 for (i = 1; i < cf->args->nelts; i++) {
297 cv = ngx_array_push(*a);
298 if (cv == NULL) {
299 return NGX_CONF_ERROR;
300 }
301
302 ngx_memzero(&ccv, sizeof(ngx_http_compile_complex_value_t));
303
304 ccv.cf = cf;
305 ccv.value = &value[i];
306 ccv.complex_value = cv;
307
308 if (ngx_http_compile_complex_value(&ccv) != NGX_OK) {
309 return NGX_CONF_ERROR;
310 }
311 }
312
313 return NGX_CONF_OK;
314}

332ngx_int_t
333ngx_http_script_compile(ngx_http_script_compile_t *sc)
334{
335 u_char ch;
336 ngx_str_t name;
337 ngx_uint_t i, bracket;
338
339 if (ngx_http_script_init_arrays(sc) != NGX_OK) {
340 return NGX_ERROR;
341 }
342
343 for (i = 0; i < sc->source->len; /* void */ ) {
344
345 name.len = 0;
346
347 if (sc->source->data[i] == '$') {
348
349 if (++i == sc->source->len) {
350 goto invalid_variable;
351 }
352
353 if (sc->source->data[i] >= '1' && sc->source->data[i] <= '9') {
354#if (NGX_PCRE)
355 ngx_uint_t n;
356
357 n = sc->source->data[i] - '0';
358
359 if (sc->captures_mask & ((ngx_uint_t) 1 << n)) {
360 sc->dup_capture = 1;
361 }
362
363 sc->captures_mask |= (ngx_uint_t) 1 << n;
364
365 if (ngx_http_script_add_capture_code(sc, n) != NGX_OK) {
366 return NGX_ERROR;
367 }
368
369 i++;
370
371 continue;
372#else
373 ngx_conf_log_error(NGX_LOG_EMERG, sc->cf, 0,
374 "using variable \"$%c\" requires "
375 "PCRE library", sc->source->data[i]);
376 return NGX_ERROR;
377#endif
378 }
379
380 if (sc->source->data[i] == '{') {
381 bracket = 1;
382
383 if (++i == sc->source->len) {
384 goto invalid_variable;
385 }
386
387 name.data = &sc->source->data[i];
388
389 } else {
390 bracket = 0;
391 name.data = &sc->source->data[i];
392 }
393
394 for ( /* void */ ; i < sc->source->len; i++, name.len++) {
395 ch = sc->source->data[i];
396
397 if (ch == '}' && bracket) {
398 i++;
399 bracket = 0;
400 break;
401 }
402
403 if ((ch >= 'A' && ch <= 'Z')
404 || (ch >= 'a' && ch <= 'z')
405 || (ch >= '0' && ch <= '9')
406 || ch == '_')
407 {
408 continue;
409 }
410
411 break;
412 }
413
414 if (bracket) {
415 ngx_conf_log_error(NGX_LOG_EMERG, sc->cf, 0,
416 "the closing bracket in \"%V\" "
417 "variable is missing", &name);
418 return NGX_ERROR;
419 }
420
421 if (name.len == 0) {
422 goto invalid_variable;
423 }
424
425 sc->variables++;
426
427 if (ngx_http_script_add_var_code(sc, &name) != NGX_OK) {
428 return NGX_ERROR;
429 }
430
431 continue;
432 }
433
434 if (sc->source->data[i] == '?' && sc->compile_args) {
435 sc->args = 1;
436 sc->compile_args = 0;
437
438 if (ngx_http_script_add_args_code(sc) != NGX_OK) {
439 return NGX_ERROR;
440 }
441
442 i++;
443
444 continue;
445 }
446
447 name.data = &sc->source->data[i];
448
449 while (i < sc->source->len) {
450
451 if (sc->source->data[i] == '$') {
452 break;
453 }
454
455 if (sc->source->data[i] == '?') {
456
457 sc->args = 1;
458
459 if (sc->compile_args) {
460 break;
461 }
462 }
463
464 i++;
465 name.len++;
466 }
467
468 sc->size += name.len;
469
470 if (ngx_http_script_add_copy_code(sc, &name, (i == sc->source->len))
471 != NGX_OK)
472 {
473 return NGX_ERROR;
474 }
475 }
476
477 return ngx_http_script_done(sc);
478
479invalid_variable:
480
481 ngx_conf_log_error(NGX_LOG_EMERG, sc->cf, 0, "invalid variable name");
482
483 return NGX_ERROR;
484}

487u_char *
488ngx_http_script_run(ngx_http_request_t *r, ngx_str_t *value,
489 void *code_lengths, size_t len, void *code_values)
490{
491 ngx_uint_t i;
492 ngx_http_script_code_pt code;
493 ngx_http_script_len_code_pt lcode;
494 ngx_http_script_engine_t e;
495 ngx_http_core_main_conf_t *cmcf;
496
497 cmcf = ngx_http_get_module_main_conf(r, ngx_http_core_module);
498
499 for (i = 0; i < cmcf->variables.nelts; i++) {
500 if (r->variables[i].no_cacheable) {
501 r->variables[i].valid = 0;
502 r->variables[i].not_found = 0;
503 }
504 }
505
506 ngx_memzero(&e, sizeof(ngx_http_script_engine_t));
507
508 e.ip = code_lengths;
509 e.request = r;
510 e.flushed = 1;
511
512 while (*(uintptr_t *) e.ip) {
513 lcode = *(ngx_http_script_len_code_pt *) e.ip;
514 len += lcode(&e);
515 }
516
517
518 value->len = len;
519 value->data = ngx_pnalloc(r->pool, len);
520 if (value->data == NULL) {
521 return NULL;
522 }
523
524 e.ip = code_values;
525 e.pos = value->data;
526
527 while (*(uintptr_t *) e.ip) {
528 code = *(ngx_http_script_code_pt *) e.ip;
529 code((ngx_http_script_engine_t *) &e);
530 }
531
532 return e.pos;
533}

554static ngx_int_t
555ngx_http_script_init_arrays(ngx_http_script_compile_t *sc)
556{
557 ngx_uint_t n;
558
559 if (sc->flushes && *sc->flushes == NULL) {
560 n = sc->variables ? sc->variables : 1;
561 *sc->flushes = ngx_array_create(sc->cf->pool, n, sizeof(ngx_uint_t));
562 if (*sc->flushes == NULL) {
563 return NGX_ERROR;
564 }
565 }
566
567 if (*sc->lengths == NULL) {
568 n = sc->variables * (2 * sizeof(ngx_http_script_copy_code_t)
569 + sizeof(ngx_http_script_var_code_t))
570 + sizeof(uintptr_t);
571
572 *sc->lengths = ngx_array_create(sc->cf->pool, n, 1);
573 if (*sc->lengths == NULL) {
574 return NGX_ERROR;
575 }
576 }
577
578 if (*sc->values == NULL) {
579 n = (sc->variables * (2 * sizeof(ngx_http_script_copy_code_t)
580 + sizeof(ngx_http_script_var_code_t))
581 + sizeof(uintptr_t)
582 + sc->source->len
583 + sizeof(uintptr_t) - 1)
584 & ~(sizeof(uintptr_t) - 1);
585
586 *sc->values = ngx_array_create(sc->cf->pool, n, 1);
587 if (*sc->values == NULL) {
588 return NGX_ERROR;
589 }
590 }
591
592 sc->variables = 0;
593
594 return NGX_OK;
595}

598static ngx_int_t
599ngx_http_script_done(ngx_http_script_compile_t *sc)
600{
601 ngx_str_t zero;
602 uintptr_t *code;
603
604 if (sc->zero) {
605
606 zero.len = 1;
607 zero.data = (u_char *) "\0";
608
609 if (ngx_http_script_add_copy_code(sc, &zero, 0) != NGX_OK) {
610 return NGX_ERROR;
611 }
612 }
613
614 if (sc->conf_prefix || sc->root_prefix) {
615 if (ngx_http_script_add_full_name_code(sc) != NGX_OK) {
616 return NGX_ERROR;
617 }
618 }
619
620 if (sc->complete_lengths) {
621 code = ngx_http_script_add_code(*sc->lengths, sizeof(uintptr_t), NULL);
622 if (code == NULL) {
623 return NGX_ERROR;
624 }
625
626 *code = (uintptr_t) NULL;
627 }
628
629 if (sc->complete_values) {
630 code = ngx_http_script_add_code(*sc->values, sizeof(uintptr_t),
631 &sc->main);
632 if (code == NULL) {
633 return NGX_ERROR;
634 }
635
636 *code = (uintptr_t) NULL;
637 }
638
639 return NGX_OK;
640}

657void *
658ngx_http_script_add_code(ngx_array_t *codes, size_t size, void *code)
659{
660 u_char *elts, **p;
661 void *new;
662
663 elts = codes->elts;
664
665 new = ngx_array_push_n(codes, size);
666 if (new == NULL) {
667 return NULL;
668 }
669
670 if (code) {
671 if (elts != codes->elts) {
672 p = code;
673 *p += (u_char *) codes->elts - elts;
674 }
675 }
676
677 return new;
678}

681static ngx_int_t
682ngx_http_script_add_copy_code(ngx_http_script_compile_t *sc, ngx_str_t *value,
683 ngx_uint_t last)
684{
685 u_char *p;
686 size_t size, len, zero;
687 ngx_http_script_copy_code_t *code;
688
689 zero = (sc->zero && last);
690 len = value->len + zero;
691
692 code = ngx_http_script_add_code(*sc->lengths,
693 sizeof(ngx_http_script_copy_code_t), NULL);
694 if (code == NULL) {
695 return NGX_ERROR;
696 }
697
698 code->code = (ngx_http_script_code_pt) ngx_http_script_copy_len_code;
699 code->len = len;
700
701 size = (sizeof(ngx_http_script_copy_code_t) + len + sizeof(uintptr_t) - 1)
702 & ~(sizeof(uintptr_t) - 1);
703
704 code = ngx_http_script_add_code(*sc->values, size, &sc->main);
705 if (code == NULL) {
706 return NGX_ERROR;
707 }
708
709 code->code = ngx_http_script_copy_code;
710 code->len = len;
711
712 p = ngx_cpymem((u_char *) code + sizeof(ngx_http_script_copy_code_t),
713 value->data, value->len);
714
715 if (zero) {
716 *p = '\0';
717 sc->zero = 0;
718 }
719
720 return NGX_OK;
721}

727 ngx_http_script_copy_code_t *code;

732
733 return code->len;

736
737void
738ngx_http_script_copy_code(ngx_http_script_engine_t *e)
739{
740 u_char *p;
741 ngx_http_script_copy_code_t *code;
742
743 code = (ngx_http_script_copy_code_t *) e->ip;
744
745 p = e->pos;
746
747 if (!e->skip) {
748 e->pos = ngx_copy(p, e->ip + sizeof(ngx_http_script_copy_code_t),
749 code->len);
750 }
751
752 e->ip += sizeof(ngx_http_script_copy_code_t)
753 + ((code->len + sizeof(uintptr_t) - 1) & ~(sizeof(uintptr_t) - 1));
754
755 ngx_log_debug2(NGX_LOG_DEBUG_HTTP, e->request->connection->log, 0,
756 "http script copy: \"%*s\"", e->pos - p, p);
757}

760static ngx_int_t
761ngx_http_script_add_var_code(ngx_http_script_compile_t *sc, ngx_str_t *name)
762{
763 ngx_int_t index, *p;
764 ngx_http_script_var_code_t *code;
765
766 index = ngx_http_get_variable_index(sc->cf, name);
767
768 if (index == NGX_ERROR) {
769 return NGX_ERROR;
770 }
771
772 if (sc->flushes) {
773 p = ngx_array_push(*sc->flushes);
774 if (p == NULL) {
775 return NGX_ERROR;
776 }
777
778 *p = index;
779 }
780
781 code = ngx_http_script_add_code(*sc->lengths,
782 sizeof(ngx_http_script_var_code_t), NULL);
783 if (code == NULL) {
784 return NGX_ERROR;
785 }
786
787 code->code = (ngx_http_script_code_pt) ngx_http_script_copy_var_len_code;
788 code->index = (uintptr_t) index;
789
790 code = ngx_http_script_add_code(*sc->values,
791 sizeof(ngx_http_script_var_code_t),
792 &sc->main);
793 if (code == NULL) {
