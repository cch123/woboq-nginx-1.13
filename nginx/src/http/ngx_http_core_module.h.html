537#define ngx_http_clear_content_length(r) \
538 \
539 r->headers_out.content_length_n = -1; \
540 if (r->headers_out.content_length) { \
541 r->headers_out.content_length->hash = 0; \
542 r->headers_out.content_length = NULL; \
543 }

545#define ngx_http_clear_accept_ranges(r) \
546 \
547 r->allow_ranges = 0; \
548 if (r->headers_out.accept_ranges) { \
549 r->headers_out.accept_ranges->hash = 0; \
550 r->headers_out.accept_ranges = NULL; \
551 }

559 }
