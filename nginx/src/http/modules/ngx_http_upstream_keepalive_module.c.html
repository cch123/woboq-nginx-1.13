122static ngx_int_t
123ngx_http_upstream_init_keepalive(ngx_conf_t *cf,
124 ngx_http_upstream_srv_conf_t *us)
125{
126 ngx_uint_t i;
127 ngx_http_upstream_keepalive_srv_conf_t *kcf;
128 ngx_http_upstream_keepalive_cache_t *cached;
129
130 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, cf->log, 0,
131 "init keepalive");
132
133 kcf = ngx_http_conf_upstream_srv_conf(us,
134 ngx_http_upstream_keepalive_module);
135
136 if (kcf->original_init_upstream(cf, us) != NGX_OK) {
137 return NGX_ERROR;
138 }
139
140 kcf->original_init_peer = us->peer.init;
141
142 us->peer.init = ngx_http_upstream_init_keepalive_peer;
143
144 /* allocate cache items and add to free queue */
145
146 cached = ngx_pcalloc(cf->pool,
147 sizeof(ngx_http_upstream_keepalive_cache_t) * kcf->max_cached);
148 if (cached == NULL) {
149 return NGX_ERROR;
150 }
151
152 ngx_queue_init(&kcf->cache);
153 ngx_queue_init(&kcf->free);
154
155 for (i = 0; i < kcf->max_cached; i++) {
156 ngx_queue_insert_head(&kcf->free, &cached[i].queue);
157 cached[i].conf = kcf;
158 }
159
160 return NGX_OK;
161}

164static ngx_int_t
165ngx_http_upstream_init_keepalive_peer(ngx_http_request_t *r,
166 ngx_http_upstream_srv_conf_t *us)
167{
168 ngx_http_upstream_keepalive_peer_data_t *kp;
169 ngx_http_upstream_keepalive_srv_conf_t *kcf;
170
171 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, r->connection->log, 0,
172 "init keepalive peer");
173
174 kcf = ngx_http_conf_upstream_srv_conf(us,
175 ngx_http_upstream_keepalive_module);
176
177 kp = ngx_palloc(r->pool, sizeof(ngx_http_upstream_keepalive_peer_data_t));
178 if (kp == NULL) {
179 return NGX_ERROR;
180 }
181
182 if (kcf->original_init_peer(r, us) != NGX_OK) {
183 return NGX_ERROR;
184 }
185
186 kp->conf = kcf;
187 kp->upstream = r->upstream;
188 kp->data = r->upstream->peer.data;
189 kp->original_get_peer = r->upstream->peer.get;
190 kp->original_free_peer = r->upstream->peer.free;
191
192 r->upstream->peer.data = kp;
193 r->upstream->peer.get = ngx_http_upstream_get_keepalive_peer;
194 r->upstream->peer.free = ngx_http_upstream_free_keepalive_peer;
195
196#if (NGX_HTTP_SSL)
197 kp->original_set_session = r->upstream->peer.set_session;
198 kp->original_save_session = r->upstream->peer.save_session;
199 r->upstream->peer.set_session = ngx_http_upstream_keepalive_set_session;
200 r->upstream->peer.save_session = ngx_http_upstream_keepalive_save_session;
201#endif
202
203 return NGX_OK;
204}

207static ngx_int_t
208ngx_http_upstream_get_keepalive_peer(ngx_peer_connection_t *pc, void *data)
209{
210 ngx_http_upstream_keepalive_peer_data_t *kp = data;
211 ngx_http_upstream_keepalive_cache_t *item;
212
213 ngx_int_t rc;
214 ngx_queue_t *q, *cache;
215 ngx_connection_t *c;
216
217 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, pc->log, 0,
218 "get keepalive peer");
219
220 /* ask balancer */
221
222 rc = kp->original_get_peer(pc, kp->data);
223
224 if (rc != NGX_OK) {
225 return rc;
226 }
227
228 /* search cache for suitable connection */
229
230 cache = &kp->conf->cache;
231
232 for (q = ngx_queue_head(cache);
233 q != ngx_queue_sentinel(cache);
234 q = ngx_queue_next(q))
235 {
236 item = ngx_queue_data(q, ngx_http_upstream_keepalive_cache_t, queue);
237 c = item->connection;
238
239 if (ngx_memn2cmp((u_char *) &item->sockaddr, (u_char *) pc->sockaddr,
240 item->socklen, pc->socklen)
241 == 0)
242 {
243 ngx_queue_remove(q);
244 ngx_queue_insert_head(&kp->conf->free, q);
245
246 goto found;
247 }
248 }
249
250 return NGX_OK;
251
252found:
253
254 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, pc->log, 0,
255 "get keepalive peer: using connection %p", c);
256
257 c->idle = 0;
258 c->sent = 0;
259 c->log = pc->log;
260 c->read->log = pc->log;
261 c->write->log = pc->log;
262 c->pool->log = pc->log;
263
264 pc->connection = c;
265 pc->cached = 1;
266
267 return NGX_DONE;
268}

271static void
272ngx_http_upstream_free_keepalive_peer(ngx_peer_connection_t *pc, void *data,
273 ngx_uint_t state)
274{
275 ngx_http_upstream_keepalive_peer_data_t *kp = data;
276 ngx_http_upstream_keepalive_cache_t *item;
277
278 ngx_queue_t *q;
279 ngx_connection_t *c;
280 ngx_http_upstream_t *u;
281
282 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, pc->log, 0,
283 "free keepalive peer");
284
285 /* cache valid connections */
286
287 u = kp->upstream;
288 c = pc->connection;
289
290 if (state & NGX_PEER_FAILED
291 || c == NULL
292 || c->read->eof
293 || c->read->error
294 || c->read->timedout
295 || c->write->error
296 || c->write->timedout)
297 {
298 goto invalid;
299 }
300
301 if (!u->keepalive) {
302 goto invalid;
303 }
304
305 if (!u->request_body_sent) {
306 goto invalid;
307 }
308
309 if (ngx_terminate || ngx_exiting) {
310 goto invalid;
311 }
312
313 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
314 goto invalid;
315 }
316
317 ngx_log_debug1(NGX_LOG_DEBUG_HTTP, pc->log, 0,
318 "free keepalive peer: saving connection %p", c);
319
320 if (ngx_queue_empty(&kp->conf->free)) {
321
322 q = ngx_queue_last(&kp->conf->cache);
323 ngx_queue_remove(q);
324
325 item = ngx_queue_data(q, ngx_http_upstream_keepalive_cache_t, queue);
326
327 ngx_http_upstream_keepalive_close(item->connection);
328
329 } else {
330 q = ngx_queue_head(&kp->conf->free);
331 ngx_queue_remove(q);
332
333 item = ngx_queue_data(q, ngx_http_upstream_keepalive_cache_t, queue);
334 }
335
336 ngx_queue_insert_head(&kp->conf->cache, q);
337
338 item->connection = c;
339
340 pc->connection = NULL;
341
342 if (c->read->timer_set) {
343 c->read->delayed = 0;
344 ngx_del_timer(c->read);
345 }
346 if (c->write->timer_set) {
347 ngx_del_timer(c->write);
348 }
349
350 c->write->handler = ngx_http_upstream_keepalive_dummy_handler;
351 c->read->handler = ngx_http_upstream_keepalive_close_handler;
352
353 c->data = item;
354 c->idle = 1;
355 c->log = ngx_cycle->log;
356 c->read->log = ngx_cycle->log;
357 c->write->log = ngx_cycle->log;
358 c->pool->log = ngx_cycle->log;
359
360 item->socklen = pc->socklen;
361 ngx_memcpy(&item->sockaddr, pc->sockaddr, pc->socklen);
362
363 if (c->read->ready) {
364 ngx_http_upstream_keepalive_close_handler(c->read);
365 }
366
367invalid:
368
369 kp->original_free_peer(pc, kp->data, state);
370}

381static void
382ngx_http_upstream_keepalive_close_handler(ngx_event_t *ev)
383{
384 ngx_http_upstream_keepalive_srv_conf_t *conf;
385 ngx_http_upstream_keepalive_cache_t *item;
386
387 int n;
388 char buf[1];
389 ngx_connection_t *c;
390
391 ngx_log_debug0(NGX_LOG_DEBUG_HTTP, ev->log, 0,
392 "keepalive close handler");
393
394 c = ev->data;
395
396 if (c->close) {
397 goto close;
398 }
399
400 n = recv(c->fd, buf, 1, MSG_PEEK);
401
402 if (n == -1 && ngx_socket_errno == NGX_EAGAIN) {
403 ev->ready = 0;
404
405 if (ngx_handle_read_event(c->read, 0) != NGX_OK) {
406 goto close;
407 }
408
409 return;
410 }
411
412close:
413
414 item = c->data;
415 conf = item->conf;
416
417 ngx_http_upstream_keepalive_close(c);
418
419 ngx_queue_remove(&item->queue);
420 ngx_queue_insert_head(&conf->free, &item->queue);
421}

424static void
425ngx_http_upstream_keepalive_close(ngx_connection_t *c)
426{
427
428#if (NGX_HTTP_SSL)
429
430 if (c->ssl) {
431 c->ssl->no_wait_shutdown = 1;
432 c->ssl->no_send_shutdown = 1;
433
434 if (ngx_ssl_shutdown(c) == NGX_AGAIN) {
435 c->ssl->handler = ngx_http_upstream_keepalive_close;
436 return;
437 }
438 }
439
440#endif
441
442 ngx_destroy_pool(c->pool);
443 ngx_close_connection(c);
444}

470static void *
471ngx_http_upstream_keepalive_create_conf(ngx_conf_t *cf)
472{
473 ngx_http_upstream_keepalive_srv_conf_t *conf;
474
475 conf = ngx_pcalloc(cf->pool,
476 sizeof(ngx_http_upstream_keepalive_srv_conf_t));
477 if (conf == NULL) {
478 return NULL;
479 }
480
481 /*
482 * set by ngx_pcalloc():
483 *
484 * conf->original_init_upstream = NULL;
485 * conf->original_init_peer = NULL;
486 * conf->max_cached = 0;

488
489 return conf;
490}

493static char *
494ngx_http_upstream_keepalive(ngx_conf_t *cf, ngx_command_t *cmd, void *conf)
495{
496 ngx_http_upstream_srv_conf_t *uscf;
497 ngx_http_upstream_keepalive_srv_conf_t *kcf = conf;
498
499 ngx_int_t n;
500 ngx_str_t *value;
501
502 if (kcf->max_cached) {
503 return "is duplicate";
504 }
505
506 /* read options */
507
508 value = cf->args->elts;
509
510 n = ngx_atoi(value[1].data, value[1].len);
511
512 if (n == NGX_ERROR || n == 0) {
513 ngx_conf_log_error(NGX_LOG_EMERG, cf, 0,
514 "invalid value \"%V\" in \"%V\" directive",
515 &value[1], &cmd->name);
516 return NGX_CONF_ERROR;
517 }
518
519 kcf->max_cached = n;
520
521 uscf = ngx_http_conf_get_module_srv_conf(cf, ngx_http_upstream_module);
522
523 kcf->original_init_upstream = uscf->peer.init_upstream
524 ? uscf->peer.init_upstream
525 : ngx_http_upstream_init_round_robin;
526
527 uscf->peer.init_upstream = ngx_http_upstream_init_keepalive;
528
529 return NGX_CONF_OK;
530}
