73void
74ngx_master_process_cycle(ngx_cycle_t *cycle)
75{
76 char *title;
77 u_char *p;
78 size_t size;
79 ngx_int_t i;
80 ngx_uint_t n, sigio;
81 sigset_t set;
82 struct itimerval itv;
83 ngx_uint_t live;
84 ngx_msec_t delay;
85 ngx_listening_t *ls;
86 ngx_core_conf_t *ccf;
87
88 sigemptyset(&set);
89 sigaddset(&set, SIGCHLD);
90 sigaddset(&set, SIGALRM);
91 sigaddset(&set, SIGIO);
92 sigaddset(&set, SIGINT);
93 sigaddset(&set, ngx_signal_value(NGX_RECONFIGURE_SIGNAL));
94 sigaddset(&set, ngx_signal_value(NGX_REOPEN_SIGNAL));
95 sigaddset(&set, ngx_signal_value(NGX_NOACCEPT_SIGNAL));
96 sigaddset(&set, ngx_signal_value(NGX_TERMINATE_SIGNAL));
97 sigaddset(&set, ngx_signal_value(NGX_SHUTDOWN_SIGNAL));
98 sigaddset(&set, ngx_signal_value(NGX_CHANGEBIN_SIGNAL));
99
100 if (sigprocmask(SIG_BLOCK, &set, NULL) == -1) {
101 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
102 "sigprocmask() failed");
103 }
104
105 sigemptyset(&set);
106
107
108 size = sizeof(master_process);
109
110 for (i = 0; i < ngx_argc; i++) {
111 size += ngx_strlen(ngx_argv[i]) + 1;
112 }
113
114 title = ngx_pnalloc(cycle->pool, size);
115 if (title == NULL) {
116 /* fatal */
117 exit(2);
118 }
119
120 p = ngx_cpymem(title, master_process, sizeof(master_process) - 1);
121 for (i = 0; i < ngx_argc; i++) {
122 *p++ = ' ';
123 p = ngx_cpystrn(p, (u_char *) ngx_argv[i], size);
124 }
125
126 ngx_setproctitle(title);
127
128
129 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx, ngx_core_module);
130
131 ngx_start_worker_processes(cycle, ccf->worker_processes,
132 NGX_PROCESS_RESPAWN);
133 ngx_start_cache_manager_processes(cycle, 0);
134
135 ngx_new_binary = 0;
136 delay = 0;
137 sigio = 0;
138 live = 1;
139
140 for ( ;; ) {
141 if (delay) {
142 if (ngx_sigalrm) {
143 sigio = 0;
144 delay *= 2;
145 ngx_sigalrm = 0;
146 }
147
148 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, cycle->log, 0,
149 "termination cycle: %M", delay);
150
151 itv.it_interval.tv_sec = 0;
152 itv.it_interval.tv_usec = 0;
153 itv.it_value.tv_sec = delay / 1000;
154 itv.it_value.tv_usec = (delay % 1000 ) * 1000;
155
156 if (setitimer(ITIMER_REAL, &itv, NULL) == -1) {
157 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
158 "setitimer() failed");
159 }
160 }
161
162 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, cycle->log, 0, "sigsuspend");
163
164 sigsuspend(&set);
165
166 ngx_time_update();
167
168 ngx_log_debug1(NGX_LOG_DEBUG_EVENT, cycle->log, 0,
169 "wake up, sigio %i", sigio);
170
171 if (ngx_reap) {
172 ngx_reap = 0;
173 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, cycle->log, 0, "reap children");
174
175 live = ngx_reap_children(cycle);
176 }
177
178 if (!live && (ngx_terminate || ngx_quit)) {
179 ngx_master_process_exit(cycle);
180 }
181
182 if (ngx_terminate) {
183 if (delay == 0) {
184 delay = 50;
185 }
186
187 if (sigio) {
188 sigio--;
189 continue;
190 }
191
192 sigio = ccf->worker_processes + 2 /* cache processes */;
193
194 if (delay > 1000) {
195 ngx_signal_worker_processes(cycle, SIGKILL);
196 } else {
197 ngx_signal_worker_processes(cycle,
198 ngx_signal_value(NGX_TERMINATE_SIGNAL));
199 }
200
201 continue;
202 }
203
204 if (ngx_quit) {
205 ngx_signal_worker_processes(cycle,
206 ngx_signal_value(NGX_SHUTDOWN_SIGNAL));
207
208 ls = cycle->listening.elts;
209 for (n = 0; n < cycle->listening.nelts; n++) {
210 if (ngx_close_socket(ls[n].fd) == -1) {
211 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_socket_errno,
212 ngx_close_socket_n " %V failed",
213 &ls[n].addr_text);
214 }
215 }
216 cycle->listening.nelts = 0;
217
218 continue;
219 }
220
221 if (ngx_reconfigure) {
222 ngx_reconfigure = 0;
223
224 if (ngx_new_binary) {
225 ngx_start_worker_processes(cycle, ccf->worker_processes,
226 NGX_PROCESS_RESPAWN);
227 ngx_start_cache_manager_processes(cycle, 0);
228 ngx_noaccepting = 0;
229
230 continue;
231 }
232
233 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reconfiguring");
234
235 cycle = ngx_init_cycle(cycle);
236 if (cycle == NULL) {
237 cycle = (ngx_cycle_t *) ngx_cycle;
238 continue;
239 }
240
241 ngx_cycle = cycle;
242 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx,
243 ngx_core_module);
244 ngx_start_worker_processes(cycle, ccf->worker_processes,
245 NGX_PROCESS_JUST_RESPAWN);
246 ngx_start_cache_manager_processes(cycle, 1);
247
248 /* allow new processes to start */
249 ngx_msleep(100);
250
251 live = 1;
252 ngx_signal_worker_processes(cycle,
253 ngx_signal_value(NGX_SHUTDOWN_SIGNAL));
254 }
255
256 if (ngx_restart) {
257 ngx_restart = 0;
258 ngx_start_worker_processes(cycle, ccf->worker_processes,
259 NGX_PROCESS_RESPAWN);
260 ngx_start_cache_manager_processes(cycle, 0);
261 live = 1;
262 }
263
264 if (ngx_reopen) {
265 ngx_reopen = 0;
266 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reopening logs");
267 ngx_reopen_files(cycle, ccf->user);
268 ngx_signal_worker_processes(cycle,
269 ngx_signal_value(NGX_REOPEN_SIGNAL));
270 }
271
272 if (ngx_change_binary) {
273 ngx_change_binary = 0;
274 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "changing binary");
275 ngx_new_binary = ngx_exec_new_binary(cycle, ngx_argv);
276 }
277
278 if (ngx_noaccept) {
279 ngx_noaccept = 0;
280 ngx_noaccepting = 1;
281 ngx_signal_worker_processes(cycle,
282 ngx_signal_value(NGX_SHUTDOWN_SIGNAL));
283 }
284 }
285}

288void
289ngx_single_process_cycle(ngx_cycle_t *cycle)
290{
291 ngx_uint_t i;
292
293 if (ngx_set_environment(cycle, NULL) == NULL) {
294 /* fatal */
295 exit(2);
296 }
297
298 for (i = 0; cycle->modules[i]; i++) {
299 if (cycle->modules[i]->init_process) {
300 if (cycle->modules[i]->init_process(cycle) == NGX_ERROR) {
301 /* fatal */
302 exit(2);
303 }
304 }
305 }
306
307 for ( ;; ) {
308 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, cycle->log, 0, "worker cycle");
309
310 ngx_process_events_and_timers(cycle);
311
312 if (ngx_terminate || ngx_quit) {
313
314 for (i = 0; cycle->modules[i]; i++) {
315 if (cycle->modules[i]->exit_process) {
316 cycle->modules[i]->exit_process(cycle);
317 }
318 }
319
320 ngx_master_process_exit(cycle);
321 }
322
323 if (ngx_reconfigure) {
324 ngx_reconfigure = 0;
325 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reconfiguring");
326
327 cycle = ngx_init_cycle(cycle);
328 if (cycle == NULL) {
329 cycle = (ngx_cycle_t *) ngx_cycle;
330 continue;
331 }
332
333 ngx_cycle = cycle;
334 }
335
336 if (ngx_reopen) {
337 ngx_reopen = 0;
338 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reopening logs");
339 ngx_reopen_files(cycle, (ngx_uid_t) -1);
340 }
341 }
342}

345static void
346ngx_start_worker_processes(ngx_cycle_t *cycle, ngx_int_t n, ngx_int_t type)
347{
348 ngx_int_t i;
349 ngx_channel_t ch;
350
351 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "start worker processes");
352
353 ngx_memzero(&ch, sizeof(ngx_channel_t));
354
355 ch.command = NGX_CMD_OPEN_CHANNEL;
356
357 for (i = 0; i < n; i++) {
358
359 ngx_spawn_process(cycle, ngx_worker_process_cycle,
360 (void *) (intptr_t) i, "worker process", type);
361
362 ch.pid = ngx_processes[ngx_process_slot].pid;
363 ch.slot = ngx_process_slot;
364 ch.fd = ngx_processes[ngx_process_slot].channel[0];
365
366 ngx_pass_open_channel(cycle, &ch);
367 }
368}

371static void
372ngx_start_cache_manager_processes(ngx_cycle_t *cycle, ngx_uint_t respawn)
373{
374 ngx_uint_t i, manager, loader;
375 ngx_path_t **path;
376 ngx_channel_t ch;
377
378 manager = 0;
379 loader = 0;
380
381 path = ngx_cycle->paths.elts;
382 for (i = 0; i < ngx_cycle->paths.nelts; i++) {
383
384 if (path[i]->manager) {
385 manager = 1;
386 }
387
388 if (path[i]->loader) {
389 loader = 1;
390 }
391 }
392
393 if (manager == 0) {
394 return;
395 }
396
397 ngx_spawn_process(cycle, ngx_cache_manager_process_cycle,
398 &ngx_cache_manager_ctx, "cache manager process",
399 respawn ? NGX_PROCESS_JUST_RESPAWN : NGX_PROCESS_RESPAWN);
400
401 ngx_memzero(&ch, sizeof(ngx_channel_t));
402
403 ch.command = NGX_CMD_OPEN_CHANNEL;
404 ch.pid = ngx_processes[ngx_process_slot].pid;
405 ch.slot = ngx_process_slot;
406 ch.fd = ngx_processes[ngx_process_slot].channel[0];
407
408 ngx_pass_open_channel(cycle, &ch);
409
410 if (loader == 0) {
411 return;
412 }
413
414 ngx_spawn_process(cycle, ngx_cache_manager_process_cycle,
415 &ngx_cache_loader_ctx, "cache loader process",
416 respawn ? NGX_PROCESS_JUST_SPAWN : NGX_PROCESS_NORESPAWN);
417
418 ch.command = NGX_CMD_OPEN_CHANNEL;
419 ch.pid = ngx_processes[ngx_process_slot].pid;
420 ch.slot = ngx_process_slot;
421 ch.fd = ngx_processes[ngx_process_slot].channel[0];
422
423 ngx_pass_open_channel(cycle, &ch);
424}

427static void
428ngx_pass_open_channel(ngx_cycle_t *cycle, ngx_channel_t *ch)
429{
430 ngx_int_t i;
431
432 for (i = 0; i < ngx_last_process; i++) {
433
434 if (i == ngx_process_slot
435 || ngx_processes[i].pid == -1
436 || ngx_processes[i].channel[0] == -1)
437 {
438 continue;
439 }
440
441 ngx_log_debug6(NGX_LOG_DEBUG_CORE, cycle->log, 0,
442 "pass channel s:%i pid:%P fd:%d to s:%i pid:%P fd:%d",
443 ch->slot, ch->pid, ch->fd,
444 i, ngx_processes[i].pid,
445 ngx_processes[i].channel[0]);
446
447 /* TODO: NGX_AGAIN */
448
449 ngx_write_channel(ngx_processes[i].channel[0],
450 ch, sizeof(ngx_channel_t), cycle->log);
451 }
452}

455static void
456ngx_signal_worker_processes(ngx_cycle_t *cycle, int signo)
457{
458 ngx_int_t i;
459 ngx_err_t err;
460 ngx_channel_t ch;
461
462 ngx_memzero(&ch, sizeof(ngx_channel_t));
463
464#if (NGX_BROKEN_SCM_RIGHTS)
465
466 ch.command = 0;
467
468#else
469
470 switch (signo) {
471
472 case ngx_signal_value(NGX_SHUTDOWN_SIGNAL):
473 ch.command = NGX_CMD_QUIT;
474 break;
475
476 case ngx_signal_value(NGX_TERMINATE_SIGNAL):
477 ch.command = NGX_CMD_TERMINATE;
478 break;
479
480 case ngx_signal_value(NGX_REOPEN_SIGNAL):
481 ch.command = NGX_CMD_REOPEN;
482 break;
483
484 default:
485 ch.command = 0;
486 }
487
488#endif
489
490 ch.fd = -1;
491
492
493 for (i = 0; i < ngx_last_process; i++) {
494
495 ngx_log_debug7(NGX_LOG_DEBUG_EVENT, cycle->log, 0,
496 "child: %i %P e:%d t:%d d:%d r:%d j:%d",
497 i,
498 ngx_processes[i].pid,
499 ngx_processes[i].exiting,
500 ngx_processes[i].exited,
501 ngx_processes[i].detached,
502 ngx_processes[i].respawn,
503 ngx_processes[i].just_spawn);
504
505 if (ngx_processes[i].detached || ngx_processes[i].pid == -1) {
506 continue;
507 }
508
509 if (ngx_processes[i].just_spawn) {
510 ngx_processes[i].just_spawn = 0;
511 continue;
512 }
513
514 if (ngx_processes[i].exiting
515 && signo == ngx_signal_value(NGX_SHUTDOWN_SIGNAL))
516 {
517 continue;
518 }
519
520 if (ch.command) {
521 if (ngx_write_channel(ngx_processes[i].channel[0],
522 &ch, sizeof(ngx_channel_t), cycle->log)
523 == NGX_OK)
524 {
525 if (signo != ngx_signal_value(NGX_REOPEN_SIGNAL)) {
526 ngx_processes[i].exiting = 1;
527 }
528
529 continue;
530 }
531 }
532
533 ngx_log_debug2(NGX_LOG_DEBUG_CORE, cycle->log, 0,
534 "kill (%P, %d)", ngx_processes[i].pid, signo);
535
536 if (kill(ngx_processes[i].pid, signo) == -1) {
537 err = ngx_errno;
538 ngx_log_error(NGX_LOG_ALERT, cycle->log, err,
539 "kill(%P, %d) failed", ngx_processes[i].pid, signo);
540
541 if (err == NGX_ESRCH) {
542 ngx_processes[i].exited = 1;
543 ngx_processes[i].exiting = 0;
544 ngx_reap = 1;
545 }
546
547 continue;
548 }
549
550 if (signo != ngx_signal_value(NGX_REOPEN_SIGNAL)) {
551 ngx_processes[i].exiting = 1;
552 }
553 }
554}

557static ngx_uint_t
558ngx_reap_children(ngx_cycle_t *cycle)
559{
560 ngx_int_t i, n;
561 ngx_uint_t live;
562 ngx_channel_t ch;
563 ngx_core_conf_t *ccf;
564
565 ngx_memzero(&ch, sizeof(ngx_channel_t));
566
567 ch.command = NGX_CMD_CLOSE_CHANNEL;
568 ch.fd = -1;
569
570 live = 0;
571 for (i = 0; i < ngx_last_process; i++) {
572
573 ngx_log_debug7(NGX_LOG_DEBUG_EVENT, cycle->log, 0,
574 "child: %i %P e:%d t:%d d:%d r:%d j:%d",
575 i,
576 ngx_processes[i].pid,
577 ngx_processes[i].exiting,
578 ngx_processes[i].exited,
579 ngx_processes[i].detached,
580 ngx_processes[i].respawn,
581 ngx_processes[i].just_spawn);
582
583 if (ngx_processes[i].pid == -1) {
584 continue;
585 }
586
587 if (ngx_processes[i].exited) {
588
589 if (!ngx_processes[i].detached) {
590 ngx_close_channel(ngx_processes[i].channel, cycle->log);
591
592 ngx_processes[i].channel[0] = -1;
593 ngx_processes[i].channel[1] = -1;
594
595 ch.pid = ngx_processes[i].pid;
596 ch.slot = i;
597
598 for (n = 0; n < ngx_last_process; n++) {
599 if (ngx_processes[n].exited
600 || ngx_processes[n].pid == -1
601 || ngx_processes[n].channel[0] == -1)
602 {
603 continue;
604 }
605
606 ngx_log_debug3(NGX_LOG_DEBUG_CORE, cycle->log, 0,
607 "pass close channel s:%i pid:%P to:%P",
608 ch.slot, ch.pid, ngx_processes[n].pid);
609
610 /* TODO: NGX_AGAIN */
611
612 ngx_write_channel(ngx_processes[n].channel[0],
613 &ch, sizeof(ngx_channel_t), cycle->log);
614 }
615 }
616
617 if (ngx_processes[i].respawn
618 && !ngx_processes[i].exiting
619 && !ngx_terminate
620 && !ngx_quit)
621 {
622 if (ngx_spawn_process(cycle, ngx_processes[i].proc,
623 ngx_processes[i].data,
624 ngx_processes[i].name, i)
625 == NGX_INVALID_PID)
626 {
627 ngx_log_error(NGX_LOG_ALERT, cycle->log, 0,
628 "could not respawn %s",
629 ngx_processes[i].name);
630 continue;
631 }
632
633
634 ch.command = NGX_CMD_OPEN_CHANNEL;
635 ch.pid = ngx_processes[ngx_process_slot].pid;
636 ch.slot = ngx_process_slot;
637 ch.fd = ngx_processes[ngx_process_slot].channel[0];
638
639 ngx_pass_open_channel(cycle, &ch);
640
641 live = 1;
642
643 continue;
644 }
645
646 if (ngx_processes[i].pid == ngx_new_binary) {
647
648 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx,
649 ngx_core_module);
650
651 if (ngx_rename_file((char *) ccf->oldpid.data,
652 (char *) ccf->pid.data)
653 == NGX_FILE_ERROR)
654 {
655 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
656 ngx_rename_file_n " %s back to %s failed "
657 "after the new binary process \"%s\" exited",
658 ccf->oldpid.data, ccf->pid.data, ngx_argv[0]);
659 }
660
661 ngx_new_binary = 0;
662 if (ngx_noaccepting) {
663 ngx_restart = 1;
664 ngx_noaccepting = 0;
665 }
666 }
667
668 if (i == ngx_last_process - 1) {
669 ngx_last_process--;
670
671 } else {
672 ngx_processes[i].pid = -1;
673 }
674
675 } else if (ngx_processes[i].exiting || !ngx_processes[i].detached) {
676 live = 1;
677 }
678 }
679
680 return live;
681}

684static void
685ngx_master_process_exit(ngx_cycle_t *cycle)
686{
687 ngx_uint_t i;
688
689 ngx_delete_pidfile(cycle);
690
691 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "exit");
692
693 for (i = 0; cycle->modules[i]; i++) {
694 if (cycle->modules[i]->exit_master) {
695 cycle->modules[i]->exit_master(cycle);
696 }
697 }
698
699 ngx_close_listening_sockets(cycle);
700
701 /*
702 * Copy ngx_cycle->log related data to the special static exit cycle,
703 * log, and log file structures enough to allow a signal handler to log.
704 * The handler may be called when standard ngx_cycle->log allocated from
705 * ngx_cycle->pool is already destroyed.
706 */
707
708
709 ngx_exit_log = *ngx_log_get_file_log(ngx_cycle->log);
710
711 ngx_exit_log_file.fd = ngx_exit_log.file->fd;
712 ngx_exit_log.file = &ngx_exit_log_file;
713 ngx_exit_log.next = NULL;
714 ngx_exit_log.writer = NULL;
715
716 ngx_exit_cycle.log = &ngx_exit_log;
717 ngx_exit_cycle.files = ngx_cycle->files;
718 ngx_exit_cycle.files_n = ngx_cycle->files_n;
719 ngx_cycle = &ngx_exit_cycle;
720
721 ngx_destroy_pool(cycle->pool);
722
723 exit(0);
724}

728ngx_worker_process_cycle(ngx_cycle_t *cycle, void *data)
729{
730 ngx_int_t worker = (intptr_t) data;
731
732 ngx_process = NGX_PROCESS_WORKER;
733 ngx_worker = worker;
734
735 ngx_worker_process_init(cycle, worker);
736
737 ngx_setproctitle("worker process");
738
739 for ( ;; ) {
740
741 if (ngx_exiting) {
742 if (ngx_event_no_timers_left() == NGX_OK) {
743 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "exiting");
744 ngx_worker_process_exit(cycle);
745 }
746 }
747
748 ngx_log_debug0(NGX_LOG_DEBUG_EVENT, cycle->log, 0, "worker cycle");
749
750 ngx_process_events_and_timers(cycle);
751
752 if (ngx_terminate) {
753 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "exiting");
754 ngx_worker_process_exit(cycle);
755 }
756
757 if (ngx_quit) {
758 ngx_quit = 0;
759 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0,
760 "gracefully shutting down");
761 ngx_setproctitle("worker process is shutting down");
762
763 if (!ngx_exiting) {
764 ngx_exiting = 1;
765 ngx_set_shutdown_timer(cycle);
766 ngx_close_listening_sockets(cycle);
767 ngx_close_idle_connections(cycle);
768 }
769 }
770
771 if (ngx_reopen) {
772 ngx_reopen = 0;
773 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reopening logs");
774 ngx_reopen_files(cycle, -1);
775 }
776 }
777}

780static void
781ngx_worker_process_init(ngx_cycle_t *cycle, ngx_int_t worker)
782{
783 sigset_t set;
784 ngx_int_t n;
785 ngx_time_t *tp;
786 ngx_uint_t i;
787 ngx_cpuset_t *cpu_affinity;
788 struct rlimit rlmt;
789 ngx_core_conf_t *ccf;
790 ngx_listening_t *ls;
791
792 if (ngx_set_environment(cycle, NULL) == NULL) {
793 /* fatal */
794 exit(2);
795 }
796
797 ccf = (ngx_core_conf_t *) ngx_get_conf(cycle->conf_ctx, ngx_core_module);
798
799 if (worker >= 0 && ccf->priority != 0) {
800 if (setpriority(PRIO_PROCESS, 0, ccf->priority) == -1) {
801 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
802 "setpriority(%d) failed", ccf->priority);
803 }
804 }
805
806 if (ccf->rlimit_nofile != NGX_CONF_UNSET) {
807 rlmt.rlim_cur = (rlim_t) ccf->rlimit_nofile;
808 rlmt.rlim_max = (rlim_t) ccf->rlimit_nofile;
809
810 if (setrlimit(RLIMIT_NOFILE, &rlmt) == -1) {
811 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
812 "setrlimit(RLIMIT_NOFILE, %i) failed",
813 ccf->rlimit_nofile);
814 }
815 }
816
817 if (ccf->rlimit_core != NGX_CONF_UNSET) {
818 rlmt.rlim_cur = (rlim_t) ccf->rlimit_core;
819 rlmt.rlim_max = (rlim_t) ccf->rlimit_core;
820
821 if (setrlimit(RLIMIT_CORE, &rlmt) == -1) {
822 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
823 "setrlimit(RLIMIT_CORE, %O) failed",
824 ccf->rlimit_core);
825 }
826 }
827
828 if (geteuid() == 0) {
829 if (setgid(ccf->group) == -1) {
830 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
831 "setgid(%d) failed", ccf->group);
832 /* fatal */
833 exit(2);
834 }
835
836 if (initgroups(ccf->username, ccf->group) == -1) {
837 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
838 "initgroups(%s, %d) failed",
839 ccf->username, ccf->group);
840 }
841
842#if (NGX_HAVE_PR_SET_KEEPCAPS && NGX_HAVE_CAPABILITIES)
843 if (ccf->transparent && ccf->user) {
844 if (prctl(PR_SET_KEEPCAPS, 1, 0, 0, 0) == -1) {
845 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
846 "prctl(PR_SET_KEEPCAPS, 1) failed");
847 /* fatal */
848 exit(2);
849 }
850 }
851#endif
852
853 if (setuid(ccf->user) == -1) {
854 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
855 "setuid(%d) failed", ccf->user);
856 /* fatal */
857 exit(2);
858 }
859
860#if (NGX_HAVE_CAPABILITIES)
861 if (ccf->transparent && ccf->user) {
862 struct __user_cap_data_struct data;
863 struct __user_cap_header_struct header;
864
865 ngx_memzero(&header, sizeof(struct __user_cap_header_struct));
866 ngx_memzero(&data, sizeof(struct __user_cap_data_struct));
867
868 header.version = _LINUX_CAPABILITY_VERSION_1;
869 data.effective = CAP_TO_MASK(CAP_NET_RAW);
870 data.permitted = data.effective;
871
872 if (syscall(SYS_capset, &header, &data) == -1) {
873 ngx_log_error(NGX_LOG_EMERG, cycle->log, ngx_errno,
874 "capset() failed");
875 /* fatal */
876 exit(2);
877 }
878 }
879#endif
880 }
881
882 if (worker >= 0) {
883 cpu_affinity = ngx_get_cpu_affinity(worker);
884
885 if (cpu_affinity) {
886 ngx_setaffinity(cpu_affinity, cycle->log);
887 }
888 }
889
890#if (NGX_HAVE_PR_SET_DUMPABLE)
891
892 /* allow coredump after setuid() in Linux 2.4.x */
893
894 if (prctl(PR_SET_DUMPABLE, 1, 0, 0, 0) == -1) {
895 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
896 "prctl(PR_SET_DUMPABLE) failed");
897 }
898
899#endif
900
901 if (ccf->working_directory.len) {
902 if (chdir((char *) ccf->working_directory.data) == -1) {
903 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
904 "chdir(\"%s\") failed", ccf->working_directory.data);
905 /* fatal */
906 exit(2);
907 }
908 }
909
910 sigemptyset(&set);
911
912 if (sigprocmask(SIG_SETMASK, &set, NULL) == -1) {
913 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
914 "sigprocmask() failed");
915 }
916
917 tp = ngx_timeofday();
918 srandom(((unsigned) ngx_pid << 16) ^ tp->sec ^ tp->msec);
919
920 /*
921 * disable deleting previous events for the listening sockets because
922 * in the worker processes there are no events at all at this point
923 */
924 ls = cycle->listening.elts;
925 for (i = 0; i < cycle->listening.nelts; i++) {
926 ls[i].previous = NULL;
927 }
928
929 for (i = 0; cycle->modules[i]; i++) {
930 if (cycle->modules[i]->init_process) {
931 if (cycle->modules[i]->init_process(cycle) == NGX_ERROR) {
932 /* fatal */
933 exit(2);
934 }
935 }
936 }
937
938 for (n = 0; n < ngx_last_process; n++) {
939
940 if (ngx_processes[n].pid == -1) {
941 continue;
942 }
943
944 if (n == ngx_process_slot) {
945 continue;
946 }
947
948 if (ngx_processes[n].channel[1] == -1) {
949 continue;
950 }
951
952 if (close(ngx_processes[n].channel[1]) == -1) {
953 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
954 "close() channel failed");
955 }
956 }
957
958 if (close(ngx_processes[ngx_process_slot].channel[0]) == -1) {
959 ngx_log_error(NGX_LOG_ALERT, cycle->log, ngx_errno,
960 "close() channel failed");
961 }
962
963#if 0
964 ngx_last_process = 0;
965#endif
966
967 if (ngx_add_channel_event(cycle, ngx_channel, NGX_READ_EVENT,
968 ngx_channel_handler)
969 == NGX_ERROR)
970 {
971 /* fatal */
972 exit(2);
973 }
974}

977static void
978ngx_worker_process_exit(ngx_cycle_t *cycle)
979{
980 ngx_uint_t i;
981 ngx_connection_t *c;
982
983 for (i = 0; cycle->modules[i]; i++) {
984 if (cycle->modules[i]->exit_process) {
985 cycle->modules[i]->exit_process(cycle);
986 }
987 }
988
989 if (ngx_exiting) {
990 c = cycle->connections;
991 for (i = 0; i < cycle->connection_n; i++) {
992 if (c[i].fd != -1
993 && c[i].read
994 && !c[i].read->accept
995 && !c[i].read->channel
996 && !c[i].read->resolver)
997 {
998 ngx_log_error(NGX_LOG_ALERT, cycle->log, 0,
999 "*%uA open socket #%d left in connection %ui",
1000 c[i].number, c[i].fd, i);
1001 ngx_debug_quit = 1;
1002 }
1003 }
1004
1005 if (ngx_debug_quit) {
1006 ngx_log_error(NGX_LOG_ALERT, cycle->log, 0, "aborting");
1007 ngx_debug_point();
1008 }
1009 }
1010
1011 /*
1012 * Copy ngx_cycle->log related data to the special static exit cycle,
1013 * log, and log file structures enough to allow a signal handler to log.
1014 * The handler may be called when standard ngx_cycle->log allocated from
1015 * ngx_cycle->pool is already destroyed.
1016 */
1017
1018 ngx_exit_log = *ngx_log_get_file_log(ngx_cycle->log);
1019
1020 ngx_exit_log_file.fd = ngx_exit_log.file->fd;
1021 ngx_exit_log.file = &ngx_exit_log_file;
1022 ngx_exit_log.next = NULL;
1023 ngx_exit_log.writer = NULL;
1024
1025 ngx_exit_cycle.log = &ngx_exit_log;
1026 ngx_exit_cycle.files = ngx_cycle->files;
1027 ngx_exit_cycle.files_n = ngx_cycle->files_n;
1028 ngx_cycle = &ngx_exit_cycle;
1029
1030 ngx_destroy_pool(cycle->pool);
1031
1032 ngx_log_error(NGX_LOG_NOTICE, ngx_cycle->log, 0, "exit");
1033
1034 exit(0);
1035}

1038static void
1039ngx_channel_handler(ngx_event_t *ev)
1040{
1041 ngx_int_t n;
1042 ngx_channel_t ch;
1043 ngx_connection_t *c;
1044
1045 if (ev->timedout) {
1046 ev->timedout = 0;
1047 return;
1048 }
1049
1050 c = ev->data;
1051
1052 ngx_log_debug0(NGX_LOG_DEBUG_CORE, ev->log, 0, "channel handler");
1053
1054 for ( ;; ) {
1055
1056 n = ngx_read_channel(c->fd, &ch, sizeof(ngx_channel_t), ev->log);
1057
1058 ngx_log_debug1(NGX_LOG_DEBUG_CORE, ev->log, 0, "channel: %i", n);
1059
1060 if (n == NGX_ERROR) {
1061
1062 if (ngx_event_flags & NGX_USE_EPOLL_EVENT) {
1063 ngx_del_conn(c, 0);
1064 }
1065
1066 ngx_close_connection(c);
1067 return;
1068 }
1069
1070 if (ngx_event_flags & NGX_USE_EVENTPORT_EVENT) {
1071 if (ngx_add_event(ev, NGX_READ_EVENT, 0) == NGX_ERROR) {
1072 return;
1073 }
1074 }
1075
1076 if (n == NGX_AGAIN) {
1077 return;
1078 }
1079
1080 ngx_log_debug1(NGX_LOG_DEBUG_CORE, ev->log, 0,
1081 "channel command: %ui", ch.command);
1082
1083 switch (ch.command) {
1084
1085 case NGX_CMD_QUIT:
1086 ngx_quit = 1;
1087 break;
1088
1089 case NGX_CMD_TERMINATE:
1090 ngx_terminate = 1;
1091 break;
1092
1093 case NGX_CMD_REOPEN:
1094 ngx_reopen = 1;
1095 break;
1096
1097 case NGX_CMD_OPEN_CHANNEL:
1098
1099 ngx_log_debug3(NGX_LOG_DEBUG_CORE, ev->log, 0,
1100 "get channel s:%i pid:%P fd:%d",
1101 ch.slot, ch.pid, ch.fd);
1102
1103 ngx_processes[ch.slot].pid = ch.pid;
1104 ngx_processes[ch.slot].channel[0] = ch.fd;
1105 break;
1106
1107 case NGX_CMD_CLOSE_CHANNEL:
1108
1109 ngx_log_debug4(NGX_LOG_DEBUG_CORE, ev->log, 0,
1110 "close channel s:%i pid:%P our:%P fd:%d",
1111 ch.slot, ch.pid, ngx_processes[ch.slot].pid,
1112 ngx_processes[ch.slot].channel[0]);
1113
1114 if (close(ngx_processes[ch.slot].channel[0]) == -1) {
1115 ngx_log_error(NGX_LOG_ALERT, ev->log, ngx_errno,
1116 "close() channel failed");
1117 }
1118
1119 ngx_processes[ch.slot].channel[0] = -1;
1120 break;
1121 }
1122 }
1123}

1126static void
1127ngx_cache_manager_process_cycle(ngx_cycle_t *cycle, void *data)
1128{
1129 ngx_cache_manager_ctx_t *ctx = data;
1130
1131 void *ident[4];
1132 ngx_event_t ev;
1133
1134 /*
1135 * Set correct process type since closing listening Unix domain socket
1136 * in a master process also removes the Unix domain socket file.
1137 */
1138 ngx_process = NGX_PROCESS_HELPER;
1139
1140 ngx_close_listening_sockets(cycle);
1141
1142 /* Set a moderate number of connections for a helper process. */
1143 cycle->connection_n = 512;
1144
1145 ngx_worker_process_init(cycle, -1);
1146
1147 ngx_memzero(&ev, sizeof(ngx_event_t));
1148 ev.handler = ctx->handler;
1149 ev.data = ident;
1150 ev.log = cycle->log;
1151 ident[3] = (void *) -1;
1152
1153 ngx_use_accept_mutex = 0;
1154
1155 ngx_setproctitle(ctx->name);
1156
1157 ngx_add_timer(&ev, ctx->delay);
1158
1159 for ( ;; ) {
1160
1161 if (ngx_terminate || ngx_quit) {
1162 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "exiting");
1163 exit(0);
1164 }
1165
1166 if (ngx_reopen) {
1167 ngx_reopen = 0;
1168 ngx_log_error(NGX_LOG_NOTICE, cycle->log, 0, "reopening logs");
1169 ngx_reopen_files(cycle, -1);
1170 }
1171
1172 ngx_process_events_and_timers(cycle);
1173 }
1174}

1177static void
1178ngx_cache_manager_process_handler(ngx_event_t *ev)
1179{
1180 ngx_uint_t i;
1181 ngx_msec_t next, n;
1182 ngx_path_t **path;
1183
1184 next = 60 * 60 * 1000;
1185
1186 path = ngx_cycle->paths.elts;
1187 for (i = 0; i < ngx_cycle->paths.nelts; i++) {
1188
1189 if (path[i]->manager) {
1190 n = path[i]->manager(path[i]->data);
1191
1192 next = (n <= next) ? n : next;
1193
1194 ngx_time_update();
1195 }
1196 }
1197
1198 if (next == 0) {
1199 next = 1;
1200 }
1201
1202 ngx_add_timer(ev, next);
1203}
