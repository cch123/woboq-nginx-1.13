31static ngx_inline ngx_int_t
32ngx_array_init(ngx_array_t *array, ngx_pool_t *pool, ngx_uint_t n, size_t size)
33{
34 /*
35 * set "array->nelts" before "array->elts", otherwise MSVC thinks
36 * that "array->nelts" may be used without having been initialized
37 */
38
39 array->nelts = 0;
40 array->size = size;
41 array->nalloc = n;
42 array->pool = pool;
43
44 array->elts = ngx_palloc(pool, n * size);
45 if (array->elts == NULL) {
46 return NGX_ERROR;
47 }
48
49 return NGX_OK;
50}
