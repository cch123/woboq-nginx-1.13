87#define ngx_queue_split(h, q, n) \
88 (n)->prev = (h)->prev; \
89 (n)->prev->next = n; \
90 (n)->next = q; \
91 (h)->prev = (q)->prev; \
92 (h)->prev->next = h; \
93 (q)->prev = n;

96#define ngx_queue_add(h, n) \

99 (h)->prev = (n)->prev; \
100 (h)->prev->next = h;
